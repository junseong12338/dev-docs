05. Nginx лҰ¬лІ„мҠӨ н”„лЎқмӢң - нҠёлһҳн”Ҫмқҳ көҗнҶө м •лҰ¬В¶

лӮңмқҙлҸ„: Gamma

01~04м—җм„ң IP, DNS, HTTP/HTTPS, SSLмқ„ л°°мӣ м–ҙ. мқҙм ң мқҙ лӘЁл“  кұё мӢӨм ңлЎң м—°кІ°н•ҙмЈјлҠ” н•өмӢ¬ мһҘм№ҳ -- NginxлҘј нҢҢн—Өм№  кұ°м•ј.

Nginxк°Җ лӯҗлғҗВ¶

ліём§Ҳ

кі м„ұлҠҘ мӣ№ м„ңлІ„ + лҰ¬лІ„мҠӨ н”„лЎқмӢң + лЎңл“ң л°ёлҹ°м„ң. к·јлҚ° мҡ°лҰ¬н•ңн…Ң мӨ‘мҡ”н•ң кұҙ лҰ¬лІ„мҠӨ н”„лЎқмӢң м—ӯн• мқҙм•ј.

мӣ№ м„ңлІ„лЎңм„ңмқҳ NginxВ¶

кё°лҠҘ м„ӨлӘ…
м •м Ғ нҢҢмқј м„ңл№ҷ HTML, CSS, JS, мқҙлҜём§ҖлҘј м§Ғм ‘ м ңкіө
лҰ¬лІ„мҠӨ н”„лЎқмӢң нҒҙлқјмқҙм–ёнҠё мҡ”мІӯмқ„ л°ұм—”л“ң м„ңлІ„лЎң м „лӢ¬
SSL мў…лЈҢ HTTPS м•”нҳёнҷ”/ліөнҳёнҷ” мІҳлҰ¬
лЎңл“ң л°ёлҹ°мӢұ м—¬лҹ¬ л°ұм—”л“ң м„ңлІ„м—җ мҡ”мІӯ л¶„л°°

мҡ°лҰ¬ н”„лЎңм қнҠём—җм„ңлҠ”

NginxлҠ” лҰ¬лІ„мҠӨ н”„лЎқмӢң + SSL мў…лЈҢ м—ӯн• л§Ң н•ҙ. м •м Ғ нҢҢмқј м„ңл№ҷмқҖ Spring Bootк°Җ м•Ңм•„м„ң н•ҳлӢҲк№Ң.

лҰ¬лІ„мҠӨ н”„лЎқмӢңк°Җ лӯҗлғҗВ¶

лҰ¬лІ„мҠӨ н”„лЎқмӢңмқҳ ліём§Ҳ

нҒҙлқјмқҙм–ёнҠёмҷҖ л°ұм—”л“ң м„ңлІ„ мӮ¬мқҙм—җ м„ңм„ң, лҢҖмӢ  мҡ”мІӯмқ„ л°ӣм•„мЈјкі  м „лӢ¬н•ҙмЈјлҠ” мӨ‘к°„мһҗ. нҒҙлқјмқҙм–ёнҠёлҠ” Nginxл§Ң ліҙмқҙкі , л’Өм—җ лӯҗк°Җ мһҲлҠ”м§Җ лӘЁлҰ„.

н”„лЎқмӢң vs лҰ¬лІ„мҠӨ н”„лЎқмӢңВ¶

кө¬л¶„ н”„лЎқмӢң (Forward) лҰ¬лІ„мҠӨ н”„лЎқмӢң (Reverse)

мҳҲмӢң нҡҢмӮ¬ н”„лЎқмӢң м„ңлІ„ Nginx, Apache
мҡ©лҸ„ м ‘к·ј м ңм–ҙ, мәҗмӢұ лЎңл“ң л°ёлҹ°мӢұ, SSL мў…лЈҢ

  • лҰ¬лІ„мҠӨ н”„лЎқмӢң: "л’Өм—җ лӯҗк°Җ мһҲлҠ”м§Җ мҲЁкІЁмӨҳ" (нҒҙлқјмқҙм–ёнҠё вҶ’ Nginx вҶ’ л°ұм—”л“ң)
  • мҡ°лҰ¬н•ңн…Ң мӨ‘мҡ”н•ң кұҙ лҰ¬лІ„мҠӨ н”„лЎқмӢңм•ј.

SSL мў…лЈҢ (SSL Termination)В¶

SSL мў…лЈҢмқҳ ліём§Ҳ

Nginxк°Җ HTTPSлҘј л°ӣм•„м„ң ліөнҳёнҷ”н•ҳкі , л°ұм—”л“ңм—җлҠ” HTTPлЎң м „лӢ¬н•ҳлҠ” нҢЁн„ҙ. SSL мІҳлҰ¬лҘј Nginx н•ң кіім—җм„ңл§Ң н•ҳлӢҲк№Ң, л°ұм—”л“ңлҠ” SSL мӢ кІҪ м•Ҳ мҚЁлҸ„ лҗЁ.

  2. м„ұлҠҘ: л°ұм—”л“ңк°Җ м•”нҳёнҷ”/ліөнҳёнҷ” м•Ҳ н•ҙлҸ„ лҗЁ
  3. мң м—°м„ұ: л°ұм—”л“ң м„ңлІ„ м¶”к°Җ/көҗмІҙ мү¬мӣҖ

server лё”лЎқ нҢЁн„ҙВ¶

лӘЁл“  лҸ„л©”мқёмқҙ к°ҷмқҖ нҢЁн„ҙ 2к°ңм”© мҢҚмңјлЎң лҸјмһҲм–ҙ:

server {
    listen 80;
    # вҶ‘ 80лІҲ нҸ¬нҠё (HTTP)лЎң л“Өм–ҙмҳӨлҠ” мҡ”мІӯмқ„ л°ӣкІ лӢӨ

    server_name nexclass.knu10.cc;
    # вҶ‘ nexclass.knu10.cc лҸ„л©”мқёмңјлЎң мҳЁ мҡ”мІӯл§Ң мІҳлҰ¬

    return 301 https://$host$request_uri;
    # вҶ‘ HTTPSлЎң мҳҒкө¬ лҰ¬лӢӨмқҙл үнҠё
    # $host = nexclass.knu10.cc
    # $request_uri = /webhook/zoom (кІҪлЎң к·ёлҢҖлЎң мң м§Җ)
}

server {
    listen 443 ssl;
    # вҶ‘ 443лІҲ нҸ¬нҠём—җм„ң SSL(HTTPS) лӘЁл“ңлЎң л“ЈкІ лӢӨ

    server_name nexclass.knu10.cc;
    # вҶ‘ nexclass.knu10.cc лҸ„л©”мқёл§Ң мІҳлҰ¬

    # в”Җв”Җ SSL мқёмҰқм„ң м„Өм • (04мһҘ м°ёкі ) в”Җв”Җ
    ssl_certificate /etc/nginx/sslcert/knu10.cc_2025092558193.all.crt.pem;
    ssl_certificate_key /etc/nginx/sslcert/knu10.cc_2025092558193.key.pem;
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_prefer_server_ciphers on;
    ssl_ciphers HIGH:!aNULL:!MD5;

    # в”Җв”Җ лҰ¬лІ„мҠӨ н”„лЎқмӢң м„Өм • в”Җв”Җ
    location / {
        proxy_pass http://{WAS_INTERNAL_IP}:8082;
        # вҶ‘ н•өмӢ¬! мқҙ мҡ”мІӯмқ„ WAS м„ңлІ„мқҳ 8082 нҸ¬нҠёлЎң м „лӢ¬

        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        # вҶ‘ мӣҗлһҳ нҒҙлқјмқҙм–ёнҠё IPлҘј л°ұм—”л“ңм—җ м „лӢ¬

        proxy_set_header HOST $http_host;
        # вҶ‘ мӣҗлһҳ мҡ”мІӯ лҸ„л©”мқёмқ„ л°ұм—”л“ңм—җ м „лӢ¬

        proxy_set_header X-NginX-Proxy true;
        # вҶ‘ "мқҙ мҡ”мІӯмқҖ NginxлҘј кұ°міҗм„ң мҳЁ кұ°м•ј" н‘ңмӢң

        proxy_connect_timeout 1800;
        proxy_read_timeout 1800;
        proxy_send_timeout 1800;
        # вҶ‘ нғҖмһ„м•„мӣғ 1800мҙҲ (30л¶„). нҷ”мғҒк°•мқҳмІҳлҹј кёҙ м—°кІ° лҢҖл№„
    }
}

X-Forwarded-For м—Ҷмңјл©ҙ?

мҷң 30л¶„мқҙлӮҳ?

нҷ”мғҒк°•мқҳ(Zoom) кҙҖл Ё мӢңмҠӨн…ңмқҙлқј кёҙ м—°кІ°мқҙ н•„мҡ”н•  мҲҳ мһҲм–ҙ. Webhook мқ‘лӢөмқҙ мҳӨлһҳ кұёлҰ¬кұ°лӮҳ, лҢҖмҡ©лҹү лҚ°мқҙн„° мІҳлҰ¬ мӢң нғҖмһ„м•„мӣғ л°©м§Җ.

location лё”лЎқ мӢ¬нҷ”В¶

location л§Өм№ӯ к·ңм№ҷВ¶

# м •нҷ• л§Өм№ӯ (мөңмҡ°м„ )
location = /health {
    # /healthл§Ң л§Өм№ӯ
}

# м ‘л‘җмӮ¬ л§Өм№ӯ (мҡ°м„ )
location ^~ /api/ {
    # /api/лЎң мӢңмһ‘н•ҳлҠ” лӘЁл“  кІҪлЎң
}

# м •к·ңмӢқ л§Өм№ӯ
location ~ \.php$ {
    # .phpлЎң лҒқлӮҳлҠ” кІҪлЎң
}

# мқјл°ҳ м ‘л‘җмӮ¬ л§Өм№ӯ
location / {
    # мң„м—җм„ң м•Ҳ кұёлҰ° лӮҳлЁём§Җ м „л¶Җ (кё°ліёк°’)
}
кё°нҳё мқҳлҜё мҡ°м„ мҲңмң„
= м •нҷ• л§Өм№ӯ 1 (мөңкі )
^~ м ‘л‘җмӮ¬ л§Өм№ӯ (м •к·ңмӢқ м•Ҳ лҙ„) 2
~ м •к·ңмӢқ л§Өм№ӯ (лҢҖмҶҢл¬ёмһҗ кө¬л¶„) 3
~* м •к·ңмӢқ л§Өм№ӯ (лҢҖмҶҢл¬ёмһҗ л¬ҙмӢң) 4
(м—ҶмқҢ) мқјл°ҳ м ‘л‘җмӮ¬ л§Өм№ӯ 5 (мөңм Җ)

мҡ°лҰ¬ м„Өм •мқҖ?

location / -- к°ҖмһҘ лӢЁмҲңн•ң "лӮҳлЁём§Җ м „л¶Җ" л§Өм№ӯмқҙм•ј. NexClassлҠ” API м„ңлІ„лқјм„ң лӘЁл“  кІҪлЎңлҘј Spring BootлЎң л„ҳкё°л©ҙ лҸј.

м„Өм • кҙҖлҰ¬В¶

лӘ…л №м–ҙ м„ӨлӘ…
sudo nginx -t м„Өм • нҢҢмқј л¬ёлІ• кІҖмӮ¬ (мҲҳм • нӣ„ л°ҳл“ңмӢң!)
sudo nginx -s reload м„Өм • лӢӨмӢң л¶Ҳлҹ¬мҳӨкё° (л¬ҙмӨ‘лӢЁ)
sudo nginx -s stop Nginx мӨ‘м§Җ
sudo systemctl start nginx Nginx мӢңмһ‘
sudo systemctl status nginx Nginx мғҒнғң нҷ•мқё

nginx -t м•Ҳ н•ҳкі  reload н•ҳл©ҙ?

м„Өм •м—җ мҳӨнғҖ мһҲмңјл©ҙ Nginxк°Җ мЈҪмқ„ мҲҳ мһҲм–ҙ. к·ёлҹ¬л©ҙ LMS, HUB, NexClass м „л¶Җ м ‘мҶҚ л¶Ҳк°Җ. л°ҳл“ңмӢң -tлЎң л¬ёлІ• кІҖмӮ¬ лЁјм Җ!

# мҳ¬л°”лҘё мҲңм„ң
sudo nginx -t              # 1. л¬ёлІ• кІҖмӮ¬
# nginx: configuration file /etc/nginx/nginx.conf test is successful
sudo nginx -s reload       # 2. кІҖмӮ¬ нҶөкіј нӣ„ лҰ¬лЎңл“ң

нҷ•мқё л¬ём ңВ¶

Q1. лҰ¬лІ„мҠӨ н”„лЎқмӢңмҷҖ нҸ¬мӣҢл“ң н”„лЎқмӢңмқҳ н•өмӢ¬ м°ЁмқҙлҠ”?

Q2. мҡ°лҰ¬ WEB м„ңлІ„м—җм„ң nexclass.knu10.ccмҷҖ www.knu10.ccк°Җ к°ҷмқҖ IPмқёлҚ° м–ҙл–»кІҢ лӢӨлҘё л°ұм—”л“ңлЎң ліҙлӮҙм ё?

Q3. proxy_set_header X-Forwarded-Forк°Җ м—Ҷмңјл©ҙ NexClass лЎңк·ём—җ м–ҙл–Ө л¬ём ңк°Җ мғқкІЁ?

Q4. nginx -tлҘј м•Ҳ н•ҳкі  nginx -s reloadлҘј н–ҲлҠ”лҚ° м„Өм •м—җ мҳӨнғҖк°Җ мһҲм—Ҳм–ҙ. м–ҙл–Ө мқјмқҙ лІҢм–ҙм ё?

Q5. SSL мў…лЈҢ(SSL Termination)к°Җ лӯҗкі , мҷң мқҙл ҮкІҢ н•ҳлҠ” кұ°м•ј?

м •лӢө ліҙкё°

A1. нҸ¬мӣҢл“ң н”„лЎқмӢңлҠ” нҒҙлқјмқҙм–ёнҠёлҘј мҲЁкё°кі , лҰ¬лІ„мҠӨ н”„лЎқмӢңлҠ” м„ңлІ„лҘј мҲЁкІЁ. нҸ¬мӣҢл“ң н”„лЎқмӢңлҠ” нҒҙлқјмқҙм–ёнҠё м•һм—җм„ң "лӮҙк°Җ лҲ„кө°м§Җ мҲЁкІЁмӨҳ" м—ӯн• мқҙкі , лҰ¬лІ„мҠӨ н”„лЎқмӢңлҠ” м„ңлІ„ м•һм—җм„ң "л’Өм—җ лӯҗк°Җ мһҲлҠ”м§Җ мҲЁкІЁмӨҳ" м—ӯн• мқҙм•ј. мҡ°лҰ¬ NginxлҠ” лҰ¬лІ„мҠӨ н”„лЎқмӢңлЎң, нҒҙлқјмқҙм–ёнҠёлҠ” NexClassк°Җ 8082м—җм„ң лҸҢм•„к°ҖлҠ”м§Җ лӘЁлҘҙкі  443л§Ң ліҙм—¬.

A2. Nginxмқҳ server_name л””л үнӢ°лёҢлЎң кө¬л¶„н•ҙ. 4к°ң лҸ„л©”мқё(www, lms, nexclass, hub)мқҙ м „л¶Җ к°ҷмқҖ IP({WEB_SERVER_IP})лЎң DNS м—°кІ°лҸј мһҲм§Җл§Ң, HTTP мҡ”мІӯм—җлҠ” Host н—ӨлҚ”м—җ лҸ„л©”мқёмқҙ л“Өм–ҙмһҲм–ҙ. Nginxк°Җ мқҙ Host н—ӨлҚ”лҘј ліҙкі  server_nameмқҙ л§Өм№ӯлҗҳлҠ” server лё”лЎқмңјлЎң ліҙлӮҙмӨҳ. мқҙкұё к°ҖмғҒ нҳёмҠӨнҢ…(Virtual Hosting)мқҙлқјкі  н•ҙ.

A3. лӘЁл“  нҒҙлқјмқҙм–ёнҠё IPк°Җ Nginx м„ңлІ„ IP({NGINX_INTERNAL_IP})лЎң м°ҚнҳҖ. Nginxк°Җ мӨ‘к°„м—җм„ң мҡ”мІӯмқ„ лҢҖмӢ  ліҙлӮҙлӢҲк№Ң, NexClass мһ…мһҘм—җм„ңлҠ” "мҡ”мІӯмқ„ ліҙлӮё лҶҲ = Nginx"лЎң ліҙм—¬. X-Forwarded-For н—ӨлҚ”к°Җ мһҲм–ҙм•ј м§„м§ң нҒҙлқјмқҙм–ёнҠё IPлҘј м•Ң мҲҳ мһҲм–ҙ. ліҙм•Ҳ лЎңк№…, м ‘мҶҚмһҗ м¶”м Ғм—җ м№ҳлӘ…м Ғмқё л¬ём ңм•ј.

A4. Nginxк°Җ мЈҪмқ„ мҲҳ мһҲм–ҙ. м„Өм • нҢҢмқјм—җ л¬ёлІ• мҳӨлҘҳк°Җ мһҲмңјл©ҙ reload мӢң Nginx н”„лЎңм„ёмҠӨк°Җ мғҲ м„Өм •мқ„ лЎңл“ң лӘ» н•ҙ. мөңм•…мқҳ кІҪмҡ° Nginxк°Җ л©Ҳм¶”л©ҙ LMS, HUB, NexClass м „л¶Җ мҷёл¶Җ м ‘мҶҚ л¶Ҳк°Җ. к·ёлһҳм„ң л°ҳл“ңмӢң nginx -tлЎң л¬ёлІ• кІҖмӮ¬ нӣ„ reload н•ҳлҠ” кұ°м•ј.

A5. SSL мў…лЈҢлҠ” Nginxк°Җ HTTPSлҘј л°ӣм•„м„ң ліөнҳёнҷ”н•ҳкі , л°ұм—”л“ңм—җлҠ” HTTPлЎң м „лӢ¬н•ҳлҠ” нҢЁн„ҙмқҙм•ј. мқҙл ҮкІҢ н•ҳлҠ” мқҙмң : (1) SSL мқёмҰқм„ңлҘј Nginx н•ң кіім—җм„ңл§Ң кҙҖлҰ¬ вҶ’ кҙҖлҰ¬ нҺёмқҳ, (2) л°ұм—”л“ңк°Җ м•”нҳёнҷ”/ліөнҳёнҷ” м•Ҳ н•ҙлҸ„ лҗЁ вҶ’ м„ұлҠҘ, (3) л°ұм—”л“ң м„ңлІ„ м¶”к°Җ/көҗмІҙк°Җ мү¬мӣҖ вҶ’ мң м—°м„ұ. лӮҙл¶Җл§қ({WAS_INTERNAL_IP})мқҖ мҷёл¶Җм—җм„ң м ‘к·ј л¶Ҳк°Җн•ҳлӢҲ HTTPм—¬лҸ„ м•Ҳм „н•ҙ.
