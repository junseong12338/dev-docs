03. DI - мқҳмЎҙм„ұ мЈјмһ…В¶

мқҙ мһҘмқҳ лӘ©н‘ң: "DIк°Җ лӯҗм•ј?" "мҷң м§Ғм ‘ new м•Ҳ н•ҙ?"м—җ м •нҷ•нһҲ лҢҖлӢөн•  мҲҳ мһҲлӢӨ

1. мқҳмЎҙм„ұ(Dependency)мқҙ лӯҗм•јВ¶

1.1 м •мқҳВ¶

мқҳмЎҙм„ұ = Aк°Җ B м—ҶмқҙлҠ” лҸҷмһ‘н•  мҲҳ м—ҶлҠ” кҙҖкі„

мҳҲмӢң:
  мһҗлҸҷм°ЁлҠ” м—”м§„мқҙ м—Ҷмңјл©ҙ лӘ» лӢ¬лҰ°лӢӨ
  вҶ’ мһҗлҸҷм°ЁлҠ” м—”м§„м—җ "мқҳмЎҙ"н•ңлӢӨ

мҪ”л“ң:
  OrderServiceлҠ” OrderRepository м—Ҷмқҙ мЈјл¬ёмқ„ м ҖмһҘн•  мҲҳ м—ҶлӢӨ
  вҶ’ OrderServiceлҠ” OrderRepositoryм—җ "мқҳмЎҙ"н•ңлӢӨ

1.2 мҪ”л“ңлЎң ліҙл©ҙВ¶

public class OrderService {
    private OrderRepository repository;  // вҶҗ мқҙкІҢ мқҳмЎҙм„ұ

    public void createOrder(Order order) {
        repository.save(order);  // repository м—Ҷмңјл©ҙ лҸҷмһ‘ л¶Ҳк°Җ
    }
}

OrderService вҶ’ OrderRepository
"OrderServiceлҠ” OrderRepositoryм—җ мқҳмЎҙн•ңлӢӨ"

2. мқҳмЎҙм„ұмқ„ н•ҙкІ°н•ҳлҠ” 3к°Җм§Җ л°©лІ•В¶

2.1 л°©лІ• 1: м§Ғм ‘ мғқм„ұ (new)В¶

public class OrderService {
    // лӮҙк°Җ м§Ғм ‘ л§Ңл“ лӢӨ
    private OrderRepository repository = new OrderRepository();
}

л¬ём ңм җ:

  1. к°•н•ң кІ°н•© (Tight Coupling)
     вҶ’ OrderRepositoryлҘј лӢӨлҘё кө¬нҳ„мІҙлЎң л°”кҫёл Өл©ҙ?
     вҶ’ OrderService мҪ”л“ңлҘј мҲҳм •н•ҙм•ј н•ңлӢӨ
     вҶ’ "OrderRepository"лқјлҠ” кө¬мІҙ нҒҙлһҳмҠӨм—җ л¬¶м—¬ мһҲлӢӨ

  2. н…ҢмҠӨнҠё л¶Ҳк°Җ
     вҶ’ к°Җм§ң(Mock) к°қмІҙлЎң л°”кҝҖ мҲҳ м—ҶлӢӨ
     вҶ’ н•ӯмғҒ м§„м§ң DBм—җ м—°кІ°н•ҙм•ј н…ҢмҠӨнҠё к°ҖлҠҘ

  3. мӨ‘ліө мғқм„ұ
     вҶ’ м—¬лҹ¬ Serviceм—җм„ң new OrderRepository() н•ҳл©ҙ
     вҶ’ к°ҷмқҖ к°қмІҙк°Җ м—¬лҹ¬ к°ң мғқкёҙлӢӨ

2.2 л°©лІ• 2: мҷёл¶Җм—җм„ң л°ӣлҠ”лӢӨ (DI)В¶

public class OrderService {
    private OrderRepository repository;

    // мҷёл¶Җм—җм„ң л„Јм–ҙмӨҖлӢӨ (мғқм„ұмһҗлЎң)
    public OrderService(OrderRepository repository) {
        this.repository = repository;
    }
}

мһҘм җ:

  1. лҠҗмҠЁн•ң кІ°н•© (Loose Coupling)
     вҶ’ OrderServiceлҠ” "лҲ„к°Җ л„Јм–ҙмЈјл“ " л°ӣкё°л§Ң н•ҳл©ҙ лҗңлӢӨ
     вҶ’ кө¬нҳ„мІҙк°Җ л°”лҖҢм–ҙлҸ„ OrderService мҪ”л“ң мҲҳм • л¶Ҳн•„мҡ”

  2. н…ҢмҠӨнҠё мү¬мӣҖ
     вҶ’ к°Җм§ң к°қмІҙлҘј л„Јм–ҙмӨ„ мҲҳ мһҲлӢӨ
     вҶ’ new OrderService(mockRepository) вҶ’ н…ҢмҠӨнҠё к°ҖлҠҘ

  3. н•ң кіім—җм„ң кҙҖлҰ¬
     вҶ’ к°қмІҙ мғқм„ұмқ„ н•ң кіі(Spring)м—җм„ң кҙҖлҰ¬

2.3 л°©лІ• 3: Springмқҙ мһҗлҸҷмңјлЎң л„Јм–ҙмӨҖлӢӨВ¶

л°©лІ• 2лҘј Springмқҙ мһҗлҸҷмңјлЎң н•ҙмЈјлҠ” кІғ = DI (Dependency Injection)

лӮҙк°Җ н•  мқј:
  "лӮҳ мқҙкұ° н•„мҡ”н•ҙ"лқјкі  м„ м–ёл§Ң н•ңлӢӨ

Springмқҙ н•  мқј:
  м„ м–ёлҗң нғҖмһ…мқҳ Beanмқ„ м°ҫм•„м„ң л„Јм–ҙмӨҖлӢӨ

3. DI (Dependency Injection) = мқҳмЎҙм„ұ мЈјмһ…В¶

3.1 м •мқҳВ¶

DI = мқҳмЎҙм„ұ мЈјмһ…
   = к°қмІҙк°Җ н•„мҡ”лЎң н•ҳлҠ” мқҳмЎҙм„ұмқ„ мҷёл¶Җ(Spring)м—җм„ң л„Јм–ҙмЈјлҠ” кІғ

"мЈјмһ…(Injection)" = "л„Јм–ҙмӨҖлӢӨ"

лҲ„к°Җ: Spring м»Ён…Ңмқҙл„Ҳк°Җ
лӯҳ:   Bean(к°қмІҙ)мқ„
м–ҙл””м—җ: лӢӨлҘё Beanмқҙ н•„мҡ”лЎң н•ҳлҠ” кіім—җ
м–ҙл–»кІҢ: setter нҳём¶ңмқҙлӮҳ мғқм„ұмһҗ нҳём¶ңлЎң

3.2 DIмқҳ 3к°Җм§Җ л°©мӢқВ¶

DI мЈјмһ… л°©мӢқ 3к°Җм§Җ

1. мғқм„ұмһҗ мЈјмһ… (Constructor Injection) вҶ’ мғқм„ұмһҗ нҢҢлқјлҜён„°лЎң л°ӣлҠ”лӢӨ вҶ’ к°ҖмһҘ к¶ҢмһҘлҗҳлҠ” л°©мӢқ

2. Setter мЈјмһ… (Setter Injection) вҶ’ setter л©”м„ңл“ңлЎң л°ӣлҠ”лӢӨ вҶ’ XML м„Өм •м—җм„ң мЈјлЎң мӮ¬мҡ© (мҡ°лҰ¬ н”„лЎңм қнҠё)

3. н•„л“ң мЈјмһ… (Field Injection) вҶ’ @Autowired, @ResourceлЎң н•„л“ңм—җ м§Ғм ‘ мЈјмһ… вҶ’ мҪ”л“ңк°Җ к°„лӢЁн•ҳм§Җл§Ң лӢЁм җлҸ„ мһҲлӢӨ

4. мЈјмһ… л°©мӢқ мғҒм„ёВ¶

4.1 мғқм„ұмһҗ мЈјмһ…В¶

public class OrderService {
    private final OrderRepository repository;

    // мғқм„ұмһҗлЎң л°ӣлҠ”лӢӨ
    public OrderService(OrderRepository repository) {
        this.repository = repository;
    }
}

<!-- XML м„Өм • -->
<bean id="orderRepository" class="com.example.OrderRepository" />

<bean id="orderService" class="com.example.OrderService">
    <constructor-arg ref="orderRepository" />
</bean>

Springмқҙ н•ҳлҠ” мқј:
  OrderRepository repo = new OrderRepository();
  OrderService service = new OrderService(repo);  // мғқм„ұмһҗлЎң мЈјмһ…

мһҘм җ:
  - л¶ҲліҖ(final) к°ҖлҠҘ вҶ’ н•ңлІҲ л„Јмңјл©ҙ л°”кҝҖ мҲҳ м—ҶлӢӨ (м•Ҳм „)
  - н•„мҲҳ мқҳмЎҙм„ұ ліҙмһҘ вҶ’ мғқм„ұмһҗм—җ мһҲмңјлӢҲ nullмқј мҲҳ м—ҶлӢӨ
  - н…ҢмҠӨнҠё мү¬мӣҖ

лӢЁм җ:
  - мқҳмЎҙм„ұ л§Һмңјл©ҙ мғқм„ұмһҗ нҢҢлқјлҜён„°к°Җ кёём–ҙм§„лӢӨ

4.2 Setter мЈјмһ…В¶

public class SqlSessionFactoryBean {
    private DataSource dataSource;

    // setterлЎң л°ӣлҠ”лӢӨ
    public void setDataSource(DataSource dataSource) {
        this.dataSource = dataSource;
    }
}

<!-- XML м„Өм • - мҡ°лҰ¬ н”„лЎңм қнҠё л°©мӢқ -->
<bean id="sqlSession" class="org.mybatis.spring.SqlSessionFactoryBean">
    <property name="dataSource" ref="dataSource" />
</bean>

Springмқҙ н•ҳлҠ” мқј:
  SqlSessionFactoryBean sqlSession = new SqlSessionFactoryBean();
  sqlSession.setDataSource(dataSource);  // setterлЎң мЈјмһ…

н•өмӢ¬:
  <property name="dataSource"> вҶ’ setDataSource() нҳём¶ң
  <property name="url"> вҶ’ setUrl() нҳём¶ң
  <property name="interval"> вҶ’ setInterval() нҳём¶ң

  name вҶ’ "set" + мІ«кёҖмһҗ лҢҖл¬ёмһҗ = setter л©”м„ңл“ңлӘ…

мһҘм җ:
  - м„ нғқм Ғ мқҳмЎҙм„ұм—җ м Ғн•© (мһҲм–ҙлҸ„ лҗҳкі  м—Ҷм–ҙлҸ„ лҗҳлҠ” кІғ)
  - XMLм—җм„ң м§ҒкҙҖм Ғ

лӢЁм җ:
  - null к°ҖлҠҘм„ұ (setter нҳём¶ң м•Ҳ н•ҳл©ҙ null)
  - л¶ҲліҖ л¶Ҳк°Җ (лӮҳмӨ‘м—җ л°”кҝҖ мҲҳ мһҲлӢӨ)

4.3 н•„л“ң мЈјмһ…В¶

public class UserServiceImpl {
    @Resource(name = "sqlSession")
    private SqlSession sqlSession;

    // setterлҸ„ мғқм„ұмһҗлҸ„ м—ҶлӢӨ
    // Springмқҙ лҰ¬н”Ңл үм…ҳмңјлЎң м§Ғм ‘ н•„л“ңм—җ л„Јм–ҙмӨҖлӢӨ
}

Springмқҙ н•ҳлҠ” мқј:
  UserServiceImpl service = new UserServiceImpl();
  // лҰ¬н”Ңл үм…ҳмңјлЎң private н•„л“ңм—җ м§Ғм ‘ м ‘к·јн•ҙм„ң к°’мқ„ л„ЈлҠ”лӢӨ
  service.sqlSession = sqlSession;  // (лҰ¬н”Ңл үм…ҳ)

мһҘм җ:
  - мҪ”л“ңк°Җ к°ҖмһҘ к°„лӢЁ

лӢЁм җ:
  - н…ҢмҠӨнҠё м–ҙл ӨмӣҖ (лҰ¬н”Ңл үм…ҳ м—ҶмқҙлҠ” к°’мқ„ л„Јмқ„ мҲҳ м—ҶлӢӨ)
  - мқҳмЎҙм„ұмқҙ мҲЁкІЁм§„лӢӨ (мҷёл¶Җм—җм„ң лӯҗк°Җ н•„мҡ”н•ңм§Җ ліҙмқҙм§Җ м•ҠлҠ”лӢӨ)
  - л¶ҲліҖ л¶Ҳк°Җ

5. @Resource vs @AutowiredВ¶

5.1 м°ЁмқҙВ¶

@Resource vs @Autowired

@Resource (javax.annotation): вҶ’ мқҙлҰ„(name)мңјлЎң Beanмқ„ м°ҫлҠ”лӢӨ вҶ’ @Resource(name = "sqlSession") вҶ’ "sqlSession"мқҙлқјлҠ” мқҙлҰ„мқҳ Beanмқ„ л„Јм–ҙмӨҳ вҶ’ Java н‘ңмӨҖ (JSR-250)

@Autowired (Spring): вҶ’ нғҖмһ…(type)мңјлЎң Beanмқ„ м°ҫлҠ”лӢӨ вҶ’ @Autowired вҶ’ "SqlSession нғҖмһ…мқҳ Beanмқ„ м°ҫм•„м„ң л„Јм–ҙмӨҳ" вҶ’ Spring м „мҡ©

мҡ°лҰ¬ н”„лЎңм қнҠё: @ResourceлҘј мЈјлЎң мӮ¬мҡ©

5.2 кІҖмғү мҲңм„ңВ¶

@Resource:
  1. nameмңјлЎң м°ҫлҠ”лӢӨ вҶ’ @Resource(name = "sqlSession")
  2. name м—Ҷмңјл©ҙ н•„л“ңлӘ…мңјлЎң м°ҫлҠ”лӢӨ вҶ’ private SqlSession sqlSession;
  3. к·ёлһҳлҸ„ м—Ҷмңјл©ҙ нғҖмһ…мңјлЎң м°ҫлҠ”лӢӨ

@Autowired:
  1. нғҖмһ…мңјлЎң м°ҫлҠ”лӢӨ
  2. к°ҷмқҖ нғҖмһ…мқҙ м—¬лҹ¬ к°ңл©ҙ мқҙлҰ„мңјлЎң кө¬л¶„
  3. @Qualifier("мқҙлҰ„")лЎң лӘ…мӢң к°ҖлҠҘ

6. DIк°Җ м—Ҷмңјл©ҙ мғқкё°лҠ” л¬ём ң (мӢӨм „)В¶

6.1 мҡ°лҰ¬ н”„лЎңм қнҠё мҳҲмӢңВ¶

context-sqlMap.xmlм—җм„ң:

<bean id="sqlSession" class="org.mybatis.spring.SqlSessionFactoryBean">
    <property name="dataSource" ref="dataSource" />
</bean>

л§Ңм•Ҫ DI м—Ҷмқҙ мҪ”л“ңлЎң н–ҲлӢӨл©ҙ:

// лӘЁл“  Serviceм—җм„ң мқҙл ҮкІҢ н•ҙм•ј н•ңлӢӨ
public class UserServiceImpl {
    private SqlSessionFactory sqlSession;

    public UserServiceImpl() {
        BasicDataSource ds = new BasicDataSource();
        ds.setUrl("jdbc:mariadb://{DB_HOST}:3306/...");
        ds.setUsername("...");
        ds.setPassword("...");

        SqlSessionFactoryBean factory = new SqlSessionFactoryBean();
        factory.setDataSource(ds);
        this.sqlSession = factory.getObject();
    }
}

// BoardServiceImplм—җм„ңлҸ„ лҳ‘к°ҷмқҙ
// 100к°ң Serviceм—җм„ң м „л¶Җ лҳ‘к°ҷмқҙ

л¬ём ң:
  вҶ’ DB мЈјмҶҢ л°”лҖҢл©ҙ? 100кө°лҚ° лӢӨ мҲҳм •
  вҶ’ л№„л°ҖлІҲнҳё л°”лҖҢл©ҙ? 100кө°лҚ° лӢӨ мҲҳм •
  вҶ’ DataSource к°қмІҙк°Җ 100к°ң мғқм„ұ вҶ’ л©”лӘЁлҰ¬ лӮӯл№„
  вҶ’ Connection Poolмқҙ 100к°ң вҶ’ DB м—°кІ° нҸӯл°ң

    private SqlSession sqlSession;

7. DIмқҳ ліём§ҲВ¶

DIмқҳ ліём§Ҳ

DIмқҳ ліём§Ҳ = кҙҖмӢ¬мӮ¬мқҳ л¶„лҰ¬ (Separation of Concerns)

Serviceмқҳ кҙҖмӢ¬мӮ¬: л№„мҰҲлӢҲмҠӨ лЎңм§Ғ вҶ’ "мЈјл¬ёмқ„ м–ҙл–»кІҢ мІҳлҰ¬н• к№Ң" вҶ’ "м„ұм Ғмқ„ м–ҙл–»кІҢ кі„мӮ°н• к№Ң"

Serviceк°Җ мӢ кІҪ м•Ҳ мҚЁлҸ„ лҗҳлҠ” кІғ: вҶ’ DB м—°кІ°мқ„ м–ҙл–»кІҢ л§Ңл“Өк№Ң вҶ’ к°қмІҙлҘј м–ҙл–»кІҢ мғқм„ұн• к№Ң вҶ’ м–ҙл–Ө кө¬нҳ„мІҙлҘј м“ёк№Ң

мқҙ "мӢ кІҪ м•Ҳ мҚЁлҸ„ лҗҳлҠ” кІғ"мқ„ Springмқҙ лҢҖмӢ  н•ҙмЈјлҠ” кІҢ DI.

мҪ”л“ңлЎң ліҙл©ҙ: "лӮҳ sqlSession н•„мҡ”н•ҙ" (м„ м–ё) "м–ҙл–»кІҢ л§Ңл“ңлҠ”м§ҖлҠ” лӘ°лқј, к·ёлғҘ мӨҳ" (Springмқҙ мЈјмһ…) вҶ’ к№”лҒ”.

8. н•өмӢ¬ м •лҰ¬В¶

н•өмӢ¬ м •лҰ¬

мқҳмЎҙм„ұ = Aк°Җ B м—Ҷмқҙ лӘ» лҸҢм•„к°ҖлҠ” кҙҖкі„ мЈјмһ… = мҷёл¶Җм—җм„ң л„Јм–ҙмЈјлҠ” кІғ DI = мқҳмЎҙм„ұ мЈјмһ… = Springмқҙ Beanмқ„ м•Ңм•„м„ң л„Јм–ҙмӨҖлӢӨ

мЈјмһ… л°©мӢқ 3к°Җм§Җ:

  • мғқм„ұмһҗ мЈјмһ…: constructor-arg (к°ҖмһҘ к¶ҢмһҘ)
  • Setter мЈјмһ…: property (XMLм—җм„ң мЈјлЎң мӮ¬мҡ©)
  • н•„л“ң мЈјмһ…: @Resource, @Autowired (к°„нҺён•ҳм§Җл§Ң лӢЁм җ мһҲмқҢ)

@Resource: мқҙлҰ„мңјлЎң м°ҫлҠ”лӢӨ (Java н‘ңмӨҖ) @Autowired: нғҖмһ…мңјлЎң м°ҫлҠ”лӢӨ (Spring м „мҡ©)

DIмқҳ ліём§Ҳ: кҙҖмӢ¬мӮ¬мқҳ л¶„лҰ¬ вҶ’ к°қмІҙ мғқм„ұ/кҙҖлҰ¬лҠ” Springмқҙ вҶ’ л№„мҰҲлӢҲмҠӨ лЎңм§ҒмқҖ лӮҙк°Җ

лӢӨмқҢ мһҘ: XML м„Өм • мҷ„м „м •ліө вҶ’ мҡ°лҰ¬ н”„лЎңм қнҠёмқҳ XML м„Өм •мқ„ н•ң мӨ„лҸ„ л№ м§җм—Ҷмқҙ мқҪлҠ”лӢӨ
