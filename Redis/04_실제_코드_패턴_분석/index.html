/**
 * нҢҢмқј: SessionUtil.java
 * мң„м№ҳ: src/main/java/kr/co/mediopia/coss/hub/common/utils/
 *
 * м—ӯн• : HTTP м„ём…ҳкіј RedisлҘј м—°лҸҷн•ҳм—¬ лЎңк·ёмқё мғҒнғң кҙҖлҰ¬
 */
public class SessionUtil {

    /**
     * лЎңк·ёмқё м •ліҙлҘј м„ём…ҳм—җ м ҖмһҘн•ҳкі  Redisм—җ л“ұлЎқ
     *
     * н•өмӢ¬ нҸ¬мқёнҠё:
     * 1. HTTP м„ём…ҳм—җ мӮ¬мҡ©мһҗ м •ліҙ м ҖмһҘ (WAS л©”лӘЁлҰ¬)
     * 2. Redisм—җ userId вҶ’ sessionId л§Өн•‘ м ҖмһҘ (м „м—ӯ мғҒнғң)
     *
     * @param session HTTP м„ём…ҳ к°қмІҙ
     * @param info    лЎңк·ёмқё мӮ¬мҡ©мһҗ м •ліҙ
     */
    public static void setLoginInfoSession(HttpSession session, LoginInfo info) {
        // мқҙлҜё лЎңк·ёмқёлҗҳм–ҙ мһҲмңјл©ҙ л¬ҙмӢң (мӨ‘ліө м ҖмһҘ л°©м§Җ)
        if (SessionUtil.getLoginInfo(session) == null) {

            // 1. HTTP м„ём…ҳм—җ лЎңк·ёмқё м •ліҙ м ҖмһҘ
            //    вҶ’ мқҙ WAS м„ңлІ„мқҳ л©”лӘЁлҰ¬м—җл§Ң мЎҙмһ¬
            session.setAttribute(Constants.SESSION_LOGIN_INFO, info);

            // 2. Redisм—җ мӮ¬мҡ©мһҗID вҶ’ м„ём…ҳID л§Өн•‘ м ҖмһҘ
            //    вҶ’ лӘЁл“  WAS м„ңлІ„м—җм„ң м ‘к·ј к°ҖлҠҘ
            //    вҶ’ нӮӨ: "KNU10-HUB:user123"
            //    вҶ’ к°’: "JSESSIONIDк°’" (мҳҲ: "ABC123DEF456")
            RedisUtil.addKeyValue(info.getUserId(), session.getId());
        }
    }

    /**
     * лЎңк·ём•„мӣғ мІҳлҰ¬
     *
     * н•өмӢ¬ нҸ¬мқёнҠё:
     * 1. Redisм—җм„ң м„ём…ҳ м •ліҙ мӮӯм ң (ліёмқё м„ём…ҳмқј л•Ңл§Ң)
     * 2. HTTP м„ём…ҳ л¬ҙнҡЁнҷ”
     * 3. мһҗлҸҷ лЎңк·ёмқё мҝ нӮӨ мӮӯм ң
     */
    public static boolean logout(HttpSession session,
                                 HttpServletRequest request,
                                 HttpServletResponse response) {

        LoginInfo info = getLoginInfo(session);

        if(info != null) {
            // вҳ… Redisм—җм„ң м„ём…ҳ мӮӯм ң
            //    вҶ’ ліёмқё м„ём…ҳмқј л•Ңл§Ң мӮӯм ңн•ҳлҠ” мқҙмң :
            //    вҶ’ лӢӨлҘё кіім—җм„ң мқҙлҜё мғҲ м„ём…ҳмңјлЎң лЎңк·ёмқён–Ҳмқ„ мҲҳ мһҲмқҢ
            //    вҶ’ мғҲ м„ём…ҳмқ„ мӮӯм ңн•ҳл©ҙ м•Ҳ лҗҳлҜҖлЎң к°’ л№„көҗ нӣ„ мӮӯм ң
            RedisUtil.removeByKeyAndValue(info.getUserId(), session.getId());
        }

        // м„ём…ҳм—җм„ң лӘЁл“  мҶҚм„ұ м ңкұ°
        session.removeAttribute(Constants.SESSION_SUBJECT_SCHEDULE_INFO);
        session.removeAttribute(Constants.SESSION_SUBJECT_SCHEDULE_STEP01);
        session.removeAttribute(Constants.SESSION_SUBJECT_SCHEDULE_STEP02);
        session.removeAttribute(Constants.SESSION_SUBJECT_SCHEDULE_STEP03);
        session.removeAttribute(Constants.SESSION_LOGIN_INFO);

        // HTTP м„ём…ҳ л¬ҙнҡЁнҷ”
        session.invalidate();

        // мһҗлҸҷ лЎңк·ёмқё мҝ нӮӨ мӮӯм ң
        Cookie cookie = HttpUtils.getCookie(request, Constants.AUTO_LOGIN_SESSION_ID);
        if (cookie != null) {
            cookie.setMaxAge(0);  // мҝ нӮӨ мҰүмӢң л§ҢлЈҢ
            response.addCookie(cookie);
            return true;
        }
        return false;
    }
}

1.3 RedisUtil.java н•өмӢ¬ мҪ”л“ң л¶„м„қ¶

/**
 * нҢҢмқј: RedisUtil.java
 * мң„м№ҳ: src/main/java/kr/co/mediopia/coss/hub/common/utils/
 *
 * м—ӯн• : Redis м—°мӮ°мқ„ мң„н•ң м •м Ғ мң нӢёлҰ¬нӢ° нҒҙлһҳмҠӨ
 *
 * нҠ№мқҙмӮ¬н•ӯ: Static л©”м„ңл“ңлЎң кө¬нҳ„
 * вҶ’ м–ҙл””м„ңл“  RedisUtil.xxx() нҳ•нғңлЎң нҳём¶ң к°ҖлҠҘ
 * вҶ’ Spring Beanмқҙм§Җл§Ң static н•„л“ң/л©”м„ңл“ң мӮ¬мҡ©
 */
@Component
public class RedisUtil {

    // ===== Static н•„л“ң м„Өм • нҢЁн„ҙ =====
    // мқјл°ҳм ҒмңјлЎң к¶ҢмһҘлҗҳм§Җ м•Ҡм§Җл§Ң, мң нӢёлҰ¬нӢ° нҒҙлһҳмҠӨм—җм„ң к°„нҳ№ мӮ¬мҡ©

    private static RedisTemplate<String, String> redisTemplate;
    public static String REDIS_KEY_PREFIX;

    // @ValueлҠ” мқёмҠӨн„ҙмҠӨ л©”м„ңл“ңм—җм„ңл§Ң лҸҷмһ‘
    // вҶ’ setterлҘј нҶөн•ҙ static н•„л“ңм—җ к°’ н• лӢ№
    @Value("${spring.redis.key-prefix}")
    public void setREDIS_KEY_PREFIX(String prefix) {
        REDIS_KEY_PREFIX = prefix;
    }

    // мғқм„ұмһҗ мЈјмһ…мңјлЎң static н•„л“ң мҙҲкё°нҷ”
    @Autowired
    public RedisUtil(RedisTemplate<String, String> redisTemplate) {
        RedisUtil.redisTemplate = redisTemplate;
    }

    // ===== н•өмӢ¬ л©”м„ңл“ң =====

    /**
     * Key-Value м ҖмһҘ (24мӢңк°„ TTL)
     *
     * мӮ¬мҡ© мҳҲмӢң:
     *   addKeyValue("user123", "sess_abc")
     *
     * Redis кІ°кіј:
     *   Key: "KNU10-HUB-TEST:user123"
     *   Value: "sess_abc"
     *   TTL: 86400мҙҲ (24мӢңк°„)
     */
    public static void addKeyValue(String key, String value) {
        // ValueOperations: String нғҖмһ… м—°мӮ° мқён„°нҺҳмқҙмҠӨ
        ValueOperations<String, String> vop = redisTemplate.opsForValue();

        // нӮӨ м•һм—җ м ‘л‘җм–ҙ м¶”к°Җн•ҳм—¬ м ҖмһҘ
        // мҳҲ: "KNU10-HUB-TEST:" + "user123" = "KNU10-HUB-TEST:user123"
        vop.set(REDIS_KEY_PREFIX + key, value);

        // TTL м„Өм •: 24мӢңк°„ нӣ„ мһҗлҸҷ мӮӯм ң
        redisTemplate.expire(REDIS_KEY_PREFIX + key, 24, TimeUnit.HOURS);
    }

    /**
     * KeyлЎң Value мЎ°нҡҢ (мҠ¬лқјмқҙл”© м„ём…ҳ)
     *
     * мҠ¬лқјмқҙл”© м„ём…ҳмқҙлһҖ?
     * вҶ’ мЎ°нҡҢн•  л•Ңл§ҲлӢӨ TTLмқ„ к°ұмӢ н•ҳм—¬ л§ҢлЈҢ мӢңк°„ м—°мһҘ
     * вҶ’ нҷңлҸҷ мӨ‘мқё мӮ¬мҡ©мһҗлҠ” лЎңк·ём•„мӣғлҗҳм§Җ м•ҠмқҢ
     *
     * @return к°’мқҙ мһҲмңјл©ҙ н•ҙлӢ№ к°’, м—Ҷмңјл©ҙ "empty"
     */
    public static String getValueByKey(String key) {
        ValueOperations<String, String> vop = redisTemplate.opsForValue();

        // null мІҙнҒ¬мҷҖ кё°ліёк°’ м„Өм •
        String retVal = StringUtil.nvl(vop.get(REDIS_KEY_PREFIX + key), "empty");

        // к°’мқҙ мЎҙмһ¬н•ҳл©ҙ TTL к°ұмӢ  (мҠ¬лқјмқҙл”© м„ём…ҳ)
        if(!retVal.equals("empty")) {
            redisTemplate.expire(REDIS_KEY_PREFIX + key, 24, TimeUnit.HOURS);
        }

        return retVal;
    }

    /**
     * KeyмҷҖ Valueк°Җ мқјм№ҳн•  л•Ңл§Ң мӮӯм ң
     *
     * мҷң лӢЁмҲң мӮӯм ңк°Җ м•„лӢҢк°Җ?
     *
     * мӢңлӮҳлҰ¬мҳӨ:
     * 1. мӮ¬мҡ©мһҗ Aк°Җ PCм—җм„ң лЎңк·ёмқё (sess_111)
     * 2. мӮ¬мҡ©мһҗ Aк°Җ лӘЁл°”мқјм—җм„ң лЎңк·ёмқё (sess_222)
     *    вҶ’ Redis: user_A = sess_222 (лҚ®м–ҙм”Җ)
     * 3. PCм—җм„ң лЎңк·ём•„мӣғ мӢңлҸ„
     *    вҶ’ лӢЁмҲң мӮӯм ңл©ҙ: Redisм—җм„ң user_A мӮӯм ңлҗЁ
     *    вҶ’ л¬ём ң: лӘЁл°”мқј м„ём…ҳк№Ңм§Җ л¬ҙнҡЁнҷ”лҗЁ!
     *
     * н•ҙкІ°:
     *    вҶ’ мӮӯм ң м „м—җ к°’ л№„көҗ
     *    вҶ’ sess_111 вү  sess_222 мқҙлҜҖлЎң мӮӯм ң м•Ҳ н•Ё
     *    вҶ’ лӘЁл°”мқј м„ём…ҳмқҖ мң м§ҖлҗЁ
     */
    public static void removeByKeyAndValue(String key, String value) {
        ValueOperations<String, String> vop = redisTemplate.opsForValue();
        String redisVal = StringUtil.nvl(vop.get(REDIS_KEY_PREFIX + key), "empty");

        // м ҖмһҘлҗң к°’кіј мқјм№ҳн•  л•Ңл§Ң мӮӯм ң
        if(redisVal.equals(value)) {
            redisTemplate.delete(REDIS_KEY_PREFIX + key);
        }
    }
}

/**
 * нҢҢмқј: LoginDupCheckInterceptor.java
 * мң„м№ҳ: src/main/java/kr/co/mediopia/coss/hub/common/interceptor/
 *
 * м—ӯн• : лӘЁл“  HTTP мҡ”мІӯм—җм„ң мӨ‘ліө лЎңк·ёмқё мІҙнҒ¬
 *
 * Spring InterceptorлһҖ?
 * вҶ’ Controller мӢӨн–ү м „/нӣ„м—җ кіөнҶө лЎңм§Ғ мІҳлҰ¬
 * вҶ’ н•„н„°мҷҖ мң мӮ¬н•ҳм§Җл§Ң Spring Context лӮҙм—җм„ң лҸҷмһ‘
 */
@Component
public class LoginDupCheckInterceptor implements HandlerInterceptor {

    /**
     * Controller мӢӨн–ү м „ нҳём¶ң
     *
     * @return true: кі„мҶҚ м§„н–ү, false: мҡ”мІӯ мӨ‘лӢЁ
     */
    @Override
    public boolean preHandle(HttpServletRequest request,
                             HttpServletResponse response,
                             Object handler) throws Exception {

        // ========================================
        // Case 1: лЎңк·ёмқё м•Ҳ лҗң мӮ¬мҡ©мһҗ вҶ’ нҶөкіј
        // ========================================
        if (!SessionUtil.isLogin(request)) {
            return true;  // кІҖмӮ¬ лҢҖмғҒ м•„лӢҳ
        }

        // ========================================
        // Case 2: лЎңк·ёмқёлҗң мӮ¬мҡ©мһҗ вҶ’ мӨ‘ліө мІҙнҒ¬
        // ========================================

        HttpSession session = request.getSession();
        LoginInfo loginInfo = SessionUtil.getLoginInfo(session);

        // Redisм—җм„ң нҳ„мһ¬ мң нҡЁн•ң м„ём…ҳ ID мЎ°нҡҢ
        String sessionId = RedisUtil.getValueByKey(loginInfo.getUserId());

        // ========================================
        // м„ём…ҳ мң нҡЁм„ұ кІҖмӮ¬
        // ========================================

        // Case 2-1: Redisм—җ м„ём…ҳ м •ліҙ м—ҶмқҢ (л§ҢлЈҢ лҳҗлҠ” л№„м •мғҒ)
        // Case 2-2: Redis м„ём…ҳкіј нҳ„мһ¬ м„ём…ҳ л¶Ҳмқјм№ҳ (лӢӨлҘё кіім—җм„ң лЎңк·ёмқё)
        if(sessionId.equals("empty") || !sessionId.equals(session.getId())) {

            // лЎңк·ём•„мӣғ мІҳлҰ¬
            SessionUtil.logout(session, request, response);

            // л©”мқё нҺҳмқҙм§ҖлЎң лҰ¬лӢӨмқҙл үнҠё
            response.sendRedirect("/");

            return false;  // мҡ”мІӯ мӨ‘лӢЁ
        }

        // Case 2-3: м •мғҒ вҶ’ кі„мҶҚ м§„н–ү
        return true;
    }

    // postHandle, afterCompletionмқҖ кё°ліё кө¬нҳ„ мӮ¬мҡ©
}

/**
 * нҢҢмқј: RedisComponent.java
 * мң„м№ҳ: src/main/java/egovframework/mediopia/lxp/common/comm/util/redis/
 *
 * м—ӯн• : лӘЁл“  лӘЁл“Ҳм—җм„ң мӮ¬мҡ©н•ҳлҠ” кіөнҶө Redis м»ҙнҸ¬л„ҢнҠё
 *
 * нҠ№м§•:
 * - лӘЁл“ҲлӘ…мқ„ нҢҢлқјлҜён„°лЎң л°ӣм•„ нӮӨ л„Өмһ„мҠӨнҺҳмқҙмҠӨ л¶„лҰ¬
 * - лӢӨм–‘н•ң лҚ°мқҙн„° нғҖмһ… м§Җмӣҗ (String, Hash, List)
 * - @ResourceлҘј мӮ¬мҡ©н•ң нҠ№м • нғҖмһ… Operations мЈјмһ…
 */
@Component
public class RedisComponent {

    // RedisTemplate мЈјмһ… (мқјл°ҳ м—°мӮ°мҡ©)
    @Autowired
    private RedisTemplate<String, Object> redisTemplate;

    // ===== нғҖмһ…лі„ Operations мЈјмһ… =====
    // @Resource: мқҙлҰ„ кё°л°ҳ мЈјмһ… (Springмқҳ @AutowiredмҷҖ лӢӨлҰ„)

    // л¬ёмһҗм—ҙ м—°мӮ°
    @Resource(name = "redisTemplate")
    private ValueOperations<String, String> stringValOps;

    // лҰ¬мҠӨнҠё м—°мӮ°
    @Resource(name = "redisTemplate")
    private ListOperations<String, Object> listValOps;

    // н•ҙмӢң м—°мӮ°
    @Resource(name = "redisTemplate")
    private HashOperations<String, String, Object> hashValOps;

    // мғҒмҲҳ
    private static String REDIS_SITE = Constants.REDIS_SITE;        // "KNU10-LMS"
    private static long REDIS_DEFULT_EXPIRE = Constants.REDIS_DEFULT_EXPIRE;  // 3600

    // ===== нӮӨ мғқм„ұ =====

    /**
     * лӘЁл“Ҳлі„ нӮӨ мғқм„ұ
     *
     * нҳ•мӢқ: {мӮ¬мқҙнҠё}:{лӘЁл“Ҳ}:{нӮӨ}
     *
     * мҳҲмӢң:
     *   parseKey("user123_course456", "lesson")
     *   вҶ’ "KNU10-LMS:lesson:user123_course456"
     *
     * мһҘм җ:
     * - лӘЁл“Ҳлі„ лҚ°мқҙн„° кІ©лҰ¬
     * - Redis CLIм—җм„ң нҢЁн„ҙ кІҖмғү мҡ©мқҙ
     * - нӮӨ м¶©лҸҢ л°©м§Җ
     */
    public String parseKey(String key, String moduleName) {
        return String.format("%s:%s:%s", REDIS_SITE, moduleName, key);
    }

    // ===== String м—°мӮ° =====

    /**
     * л¬ёмһҗм—ҙ м ҖмһҘ
     *
     * @param key        м ҖмһҘн•  нӮӨ
     * @param value      м ҖмһҘн•  к°’
     * @param expire     л§ҢлЈҢ мӢңк°„ (мҙҲ)
     * @param moduleName лӘЁл“ҲлӘ…
     */
    public void setString(String key, String value, long expire, String moduleName) {
        stringValOps.set(
            parseKey(key, moduleName),
            value,
            expire,
            TimeUnit.SECONDS
        );
    }

    /**
     * л¬ёмһҗм—ҙ м ҖмһҘ (кё°ліё TTL мӮ¬мҡ©)
     */
    public void setString(String key, String value, String moduleName) {
        stringValOps.set(
            parseKey(key, moduleName),
            value,
            REDIS_DEFULT_EXPIRE,
            TimeUnit.SECONDS
        );
    }

    /**
     * л¬ёмһҗм—ҙ мЎ°нҡҢ
     */
    public String getString(String key, String moduleName) {
        try {
            return (String) stringValOps.get(parseKey(key, moduleName));
        } catch (Exception e) {
            return null;  // мҳҲмҷё мӢң null л°ҳнҷҳ (мЎ°мҡ©н•ң мӢӨнҢЁ)
        }
    }

    // ===== Hash м—°мӮ° =====

    /**
     * н•ҙмӢң м ҖмһҘ
     *
     * мӮ¬мҡ© мҳҲмӢң:
     *   Map<String, Object> userInfo = new HashMap<>();
     *   userInfo.put("name", "нҷҚкёёлҸҷ");
     *   userInfo.put("score", 95);
     *   putHash("exam123_user456", userInfo, 3600, "exam");
     *
     * Redis кІ°кіј:
     *   Key: "KNU10-LMS:exam:exam123_user456"
     *   Fields:
     *     name: "нҷҚкёёлҸҷ"
     *     score: 95
     */
    public void putHash(String key, Map<String, Object> value,
                        long expire, String moduleName) {
        hashValOps.putAll(parseKey(key, moduleName), value);
        redisTemplate.expire(parseKey(key, moduleName), expire, TimeUnit.SECONDS);
    }

    /**
     * н•ҙмӢң нҠ№м • н•„л“ң мЎ°нҡҢ
     */
    public Object getHash(String key, String hashKey, String moduleName) {
        try {
            return hashValOps.get(parseKey(key, moduleName), hashKey);
        } catch (Exception e) {
            return null;
        }
    }

    /**
     * н•ҙмӢң н•„л“ң лӘ©лЎқ мЎ°нҡҢ
     */
    public Set<String> getHashKeys(String key, String moduleName) {
        try {
            return hashValOps.keys(parseKey(key, moduleName));
        } catch (Exception e) {
            return null;
        }
    }

    /**
     * н•ҙмӢң нҠ№м • н•„л“ң мӮӯм ң
     */
    public void deleteHash(String key, String moduleName, Object... keys) {
        hashValOps.delete(parseKey(key, moduleName), keys);
    }

    // ===== List м—°мӮ° =====

    /**
     * лҰ¬мҠӨнҠём—җ м¶”к°Җ (мҷјмӘҪ)
     *
     * мӮ¬мҡ© мҳҲмӢң:
     *   pushList("notifications_user123", "мғҲ м•ҢлҰј!", 3600, "user");
     *
     * Redis кІ°кіј:
     *   Key: "KNU10-LMS:user:notifications_user123"
     *   Value: ["мғҲ м•ҢлҰј!", "мқҙм „ м•ҢлҰј1", "мқҙм „ м•ҢлҰј2", ...]
     */
    public void pushList(String key, Object value, long expire, String moduleName) {
        listValOps.leftPush(parseKey(key, moduleName), value);
        redisTemplate.expire(parseKey(key, moduleName), expire, TimeUnit.SECONDS);
    }

    /**
     * лҰ¬мҠӨнҠё м „мІҙ мЎ°нҡҢ
     */
    public List<Object> getList(String key, String moduleName) {
        try {
            // 0л¶Җн„° -1к№Ңм§Җ = м „мІҙ мЎ°нҡҢ
            return listValOps.range(parseKey(key, moduleName), 0, -1);
        } catch (Exception e) {
            return null;
        }
    }

    // ===== кіөнҶө м—°мӮ° =====

    /**
     * нӮӨ мЎҙмһ¬ м—¬л¶Җ нҷ•мқё
     */
    public boolean isKey(String key, String moduleName) {
        return redisTemplate.hasKey(parseKey(key, moduleName));
    }

    /**
     * TTL мЎ°нҡҢ
     */
    public long getExpire(String key, String moduleName) {
        return redisTemplate.getExpire(parseKey(key, moduleName));
    }

    /**
     * нӮӨ мӮӯм ң
     */
    public void delete(String key, String moduleName) {
        redisTemplate.delete(parseKey(key, moduleName));
    }

    /**
     * TTL к°ұмӢ 
     */
    public void refreshExpire(String key, long expire, String moduleName) {
        redisTemplate.expire(parseKey(key, moduleName), expire, TimeUnit.SECONDS);
    }
}

/**
 * нҢҢмқј: ExamRedisSubscriber.java
 * мң„м№ҳ: src/main/java/.../lms/core/exam/service/redis/pubsub/
 *
 * м—ӯн• : Redis мұ„л„җм—җм„ң мӢңн—ҳ мқҙлІӨнҠё мҲҳмӢ  л°Ҹ мІҳлҰ¬

 * - Redis л©”мӢңм§ҖлҘј мҲҳмӢ н•ҳкё° мң„н•ң н‘ңмӨҖ мқён„°нҺҳмқҙмҠӨ
 * - onMessage() л©”м„ңл“ң кө¬нҳ„ н•„мҡ”
 */
@Service
public class ExamRedisSubscriber implements MessageListener {

    @Autowired
    private RedisTemplate<String, Object> redisTemplate;

    @Autowired
    private ExamSseService examSseService;  // SSE м„ңл№„мҠӨ (лёҢлқјмҡ°м Җ мӢӨмӢңк°„ м•ҢлҰј)

    /**
     * Redis л©”мӢңм§Җ мҲҳмӢ  мҪңл°ұ
     *
     * мқҙ л©”м„ңл“ңлҠ” Redisм—җм„ң л©”мӢңм§ҖлҘј мҲҳмӢ н•  л•Ң мһҗлҸҷмңјлЎң нҳём¶ңлҗЁ
     *
     * @param message Redis л©”мӢңм§Җ к°қмІҙ (л°”мқҙнҠё л°°м—ҙ)
     * @param pattern мұ„л„җ нҢЁн„ҙ (нҢЁн„ҙ кө¬лҸ… мӢң мӮ¬мҡ©)
     */
    @Override
    public void onMessage(Message message, byte[] pattern) {

        // л©”мӢңм§Җ ліёл¬ёмқҙ мһҲлҠ”м§Җ нҷ•мқё
        if(redisTemplate.getValueSerializer().deserialize(message.getBody()) != null) {

            // ===== л©”мӢңм§Җ м—ӯм§Ғл ¬нҷ” =====
            // л°”мқҙнҠё л°°м—ҙ вҶ’ Java к°қмІҙлЎң ліҖнҷҳ
            ExamPubSupVO pubSupVO = (ExamPubSupVO)
                redisTemplate.getValueSerializer().deserialize(message.getBody());

            // ===== мқҙлІӨнҠё нғҖмһ…лі„ л¶„кё° мІҳлҰ¬ =====
            switch (pubSupVO.getEventType()) {

                case "DUPLICATE_CHECK":
                    // мӢңн—ҳ мӨ‘ліө мқ‘мӢң мІҙнҒ¬
                    // вҶ’ к°ҷмқҖ н•ҷмғқмқҙ лӢӨлҘё кё°кё°м—җм„ң мӢңн—ҳ мқ‘мӢң мӢңлҸ„
                    examSseService.duplicateCheck(pubSupVO);
                    break;

                case "STOP":
                    // мӢңн—ҳ к°•м ң мў…лЈҢ
                    // вҶ’ көҗмҲҳк°Җ мӢңн—ҳ мў…лЈҢ лІ„нҠј нҒҙлҰӯ
                    // вҶ’ н•ҙлӢ№ мӢңн—ҳ мқ‘мӢң мӨ‘мқё лӘЁл“  н•ҷмғқм—җкІҢ м•ҢлҰј
                    examSseService.stopRequest(pubSupVO);
                    break;

                // н•„мҡ”мӢң м¶”к°Җ мқҙлІӨнҠё нғҖмһ… мІҳлҰ¬
                // case "PAUSE":
                //     examSseService.pauseExam(pubSupVO);
                //     break;
            }
        }
    }
}

/**
 * Redis м—җлҹ¬ мІҳлҰ¬ Best Practice
 */
@Component
public class ResilientRedisComponent {

    @Autowired
    private RedisTemplate<String, Object> redisTemplate;

    private static final Logger log = LoggerFactory.getLogger(ResilientRedisComponent.class);

    /**
     * м•Ҳм „н•ң Redis мЎ°нҡҢ
     *
     * нҠ№м§•:
     * 1. мҳҲмҷё мӢң null л°ҳнҷҳ (мһҘм•  м „нҢҢ л°©м§Җ)
     * 2. м—җлҹ¬ лЎңк№…
     * 3. нҳём¶ңл¶Җм—җм„ң null мІҙнҒ¬ н•„мҡ”
     */
    public String safeGet(String key) {
        try {
            return (String) redisTemplate.opsForValue().get(key);
        } catch (Exception e) {
            log.error("Redis GET мӢӨнҢЁ. key={}", key, e);
            return null;  // кё°ліёк°’ л°ҳнҷҳ
        }
    }

    /**
     * кё°ліёк°’ л°ҳнҷҳ нҢЁн„ҙ
     */
    public String getOrDefault(String key, String defaultValue) {
        try {
            String value = (String) redisTemplate.opsForValue().get(key);
            return value != null ? value : defaultValue;
        } catch (Exception e) {
            log.warn("Redis мЎ°нҡҢ мӢӨнҢЁ, кё°ліёк°’ л°ҳнҷҳ. key={}", key);
            return defaultValue;
        }
    }

    /**
     * Fallback нҢЁн„ҙ (Redis мӢӨнҢЁ мӢң DB мЎ°нҡҢ)
     */
    public String getWithFallback(String key, Supplier<String> dbFallback) {
        try {
            String value = (String) redisTemplate.opsForValue().get(key);
            if (value != null) {
                return value;
            }
        } catch (Exception e) {
            log.warn("Redis мЎ°нҡҢ мӢӨнҢЁ, DB Fallback мӢӨн–ү. key={}", key);
        }

        // Redis мӢӨнҢЁ лҳҗлҠ” мәҗмӢң лҜёмҠӨ вҶ’ DB мЎ°нҡҢ
        String dbValue = dbFallback.get();

        // к°ҖлҠҘн•ҳл©ҙ Redisм—җ лӢӨмӢң м ҖмһҘ
        try {
            if (dbValue != null) {
                redisTemplate.opsForValue().set(key, dbValue, 1, TimeUnit.HOURS);
            }
        } catch (Exception e) {
            log.warn("Redis м ҖмһҘ мӢӨнҢЁ. key={}", key);
        }

        return dbValue;
    }
}

@Service
public class UserService {

    @Autowired
    private ResilientRedisComponent redisComponent;

    @Autowired

        return redisComponent.getWithFallback(
            "user:" + userId + ":name",

        );
    }
}
