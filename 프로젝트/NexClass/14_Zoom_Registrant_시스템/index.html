@Slf4j
@Service
@RequiredArgsConstructor
public class RegistrantService {

    private final ZoomService zoomService;
    private final CourseUserRepository courseUserRepository;
    private final UserRepository userRepository;
    private final LessonUserRepository lessonUserRepository;

    @Value("${nexclass.dummy-email-domain}")
    private String dummyEmailDomain;   // nexclass.local

    /**
     * мҲҳк°•мғқ м „мӣҗмқ„ Zoom RegistrantлЎң л№„лҸҷкё° л“ұлЎқ
     * @param lessonCd      к°•мқҳ мҪ”л“ң
     * @param courseCd      кіјлӘ© мҪ”л“ң (мҲҳк°•мғқ лӘ©лЎқ мЎ°нҡҢмҡ©)
     * @param zoomMeetingId Zoom лҜёнҢ… ID
     */
    @Async  // лі„лҸ„ мҠӨл Ҳл“ңм—җм„ң мӢӨн–ү вҶ’ API мқ‘лӢө м§Җм—° м—ҶмқҢ
    public void registerCourseStudentsAsync(String lessonCd,
            String courseCd, String zoomMeetingId) {

        // в‘  н•ҙлӢ№ кіјлӘ©мқҳ мҲҳк°•мғқ м „мӣҗ мЎ°нҡҢ
        List<CourseUser> students =
            courseUserRepository.findByCourseCd(courseCd);
        log.info("[Async] Registrant л“ұлЎқ мӢңмһ‘: lessonCd={}, мҲҳк°•мғқ={}лӘ…",
            lessonCd, students.size());

        int successCount = 0;
        int failCount = 0;

        // в‘Ў к°Ғ мҲҳк°•мғқмқ„ Zoomм—җ л“ұлЎқ
        for (CourseUser student : students) {
            try {
                // мқҙл©”мқј: userCd@nexclass.local
                String email = student.getUserCd() + "@" + dummyEmailDomain;

                // мӮ¬мҡ©мһҗ м •ліҙ мЎ°нҡҢ (мқҙлҰ„, н•ҷлІҲ)
                User user = userRepository.findById(
                    student.getUserCd()).orElse(null);
                String userName = user != null && user.getUserName() != null
                    ? user.getUserName() : student.getUserCd();
                String userNo = user != null && user.getUserNo() != null
                    ? user.getUserNo() : student.getUserCd();

                // в‘ў Zoom Registrant API нҳём¶ң
                Map<String, Object> result = zoomService.registerParticipant(
                    zoomMeetingId, email, userName, userNo);

                // в‘Ј к°ңмқё joinUrl м¶”м¶ң + н•ңкөӯм–ҙ м„Өм •
                String joinUrl = (String) result.get("join_url");
                if (joinUrl != null && !joinUrl.contains("uiLang")) {
                    joinUrl += (joinUrl.contains("?") ? "&" : "?")
                        + "uiLang=ko-KO";
                }

                // в‘Ө LessonUser н…Ңмқҙлё”м—җ м ҖмһҘ
                LessonUser lessonUser = LessonUser.builder()
                    .lessonCd(lessonCd)
                    .userCd(student.getUserCd())
                    .registrantId((String) result.get("registrant_id"))
                    .joinUrl(joinUrl)
                    .build();
                lessonUserRepository.save(lessonUser);

                successCount++;
            } catch (Exception e) {
                failCount++;
                log.warn("[Async] Registrant л“ұлЎқ мӢӨнҢЁ: userCd={}, error={}",
                    student.getUserCd(), e.getMessage());
            }
        }
        log.info("[Async] Registrant л“ұлЎқ мҷ„лЈҢ: lessonCd={}, м„ұкіө={}, мӢӨнҢЁ={}",
            lessonCd, successCount, failCount);
    }
}

}
