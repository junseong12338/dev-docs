### 1.2 SessionUtil.java 핵심 코드 분석

```java
/**
 * 파일: SessionUtil.java
 * 위치: src/main/java/kr/co/mediopia/coss/hub/common/utils/
 *
 * 역할: HTTP 세션과 Redis를 연동하여 로그인 상태 관리
 */
public class SessionUtil {

    /**
     * 로그인 정보를 세션에 저장하고 Redis에 등록
     *
     * 핵심 포인트:
     * 1. HTTP 세션에 사용자 정보 저장 (WAS 메모리)
     * 2. Redis에 userId → sessionId 매핑 저장 (전역 상태)
     *
     * @param session HTTP 세션 객체
     * @param info    로그인 사용자 정보
     */
    public static void setLoginInfoSession(HttpSession session, LoginInfo info) {
        // 이미 로그인되어 있으면 무시 (중복 저장 방지)
        if (SessionUtil.getLoginInfo(session) == null) {

            // 1. HTTP 세션에 로그인 정보 저장
            //    → 이 WAS 서버의 메모리에만 존재
            session.setAttribute(Constants.SESSION_LOGIN_INFO, info);

            // 2. Redis에 사용자ID → 세션ID 매핑 저장
            //    → 모든 WAS 서버에서 접근 가능
            //    → 키: "KNU10-HUB:user123"
            //    → 값: "JSESSIONID값" (예: "ABC123DEF456")
            RedisUtil.addKeyValue(info.getUserId(), session.getId());
        }
    }

    /**
     * 로그아웃 처리
     *
     * 핵심 포인트:
     * 1. Redis에서 세션 정보 삭제 (본인 세션일 때만)
     * 2. HTTP 세션 무효화
     * 3. 자동 로그인 쿠키 삭제
     */
    public static boolean logout(HttpSession session,
                                 HttpServletRequest request,
                                 HttpServletResponse response) {

        LoginInfo info = getLoginInfo(session);

        if(info != null) {
            // ★ Redis에서 세션 삭제
            //    → 본인 세션일 때만 삭제하는 이유:
            //    → 다른 곳에서 이미 새 세션으로 로그인했을 수 있음
            //    → 새 세션을 삭제하면 안 되므로 값 비교 후 삭제
            RedisUtil.removeByKeyAndValue(info.getUserId(), session.getId());
        }

        // 세션에서 모든 속성 제거
        session.removeAttribute(Constants.SESSION_SUBJECT_SCHEDULE_INFO);
        session.removeAttribute(Constants.SESSION_SUBJECT_SCHEDULE_STEP01);
        session.removeAttribute(Constants.SESSION_SUBJECT_SCHEDULE_STEP02);
        session.removeAttribute(Constants.SESSION_SUBJECT_SCHEDULE_STEP03);
        session.removeAttribute(Constants.SESSION_LOGIN_INFO);

        // HTTP 세션 무효화
        session.invalidate();

        // 자동 로그인 쿠키 삭제
        Cookie cookie = HttpUtils.getCookie(request, Constants.AUTO_LOGIN_SESSION_ID);
        if (cookie != null) {
            cookie.setMaxAge(0);  // 쿠키 즉시 만료
            response.addCookie(cookie);
            return true;
        }
        return false;
    }
}
```

### 1.3 RedisUtil.java 핵심 코드 분석

```java
/**
 * 파일: RedisUtil.java
 * 위치: src/main/java/kr/co/mediopia/coss/hub/common/utils/
 *
 * 역할: Redis 연산을 위한 정적 유틸리티 클래스
 *
 * 특이사항: Static 메서드로 구현
 * → 어디서든 RedisUtil.xxx() 형태로 호출 가능
 * → Spring Bean이지만 static 필드/메서드 사용
 */
@Component
public class RedisUtil {

    // ===== Static 필드 설정 패턴 =====
    // 일반적으로 권장되지 않지만, 유틸리티 클래스에서 간혹 사용

    private static RedisTemplate<String, String> redisTemplate;
    public static String REDIS_KEY_PREFIX;

    // @Value는 인스턴스 메서드에서만 동작
    // → setter를 통해 static 필드에 값 할당
    @Value("${spring.redis.key-prefix}")
    public void setREDIS_KEY_PREFIX(String prefix) {
        REDIS_KEY_PREFIX = prefix;
    }

    // 생성자 주입으로 static 필드 초기화
    @Autowired
    public RedisUtil(RedisTemplate<String, String> redisTemplate) {
        RedisUtil.redisTemplate = redisTemplate;
    }

    // ===== 핵심 메서드 =====

    /**
     * Key-Value 저장 (24시간 TTL)
     *
     * 사용 예시:
     *   addKeyValue("user123", "sess_abc")
     *
     * Redis 결과:
     *   Key: "KNU10-HUB-TEST:user123"
     *   Value: "sess_abc"
     *   TTL: 86400초 (24시간)
     */
    public static void addKeyValue(String key, String value) {
        // ValueOperations: String 타입 연산 인터페이스
        ValueOperations<String, String> vop = redisTemplate.opsForValue();

        // 키 앞에 접두어 추가하여 저장
        // 예: "KNU10-HUB-TEST:" + "user123" = "KNU10-HUB-TEST:user123"
        vop.set(REDIS_KEY_PREFIX + key, value);

        // TTL 설정: 24시간 후 자동 삭제
        redisTemplate.expire(REDIS_KEY_PREFIX + key, 24, TimeUnit.HOURS);
    }

    /**
     * Key로 Value 조회 (슬라이딩 세션)
     *
     * 슬라이딩 세션이란?
     * → 조회할 때마다 TTL을 갱신하여 만료 시간 연장
     * → 활동 중인 사용자는 로그아웃되지 않음
     *
     * @return 값이 있으면 해당 값, 없으면 "empty"
     */
    public static String getValueByKey(String key) {
        ValueOperations<String, String> vop = redisTemplate.opsForValue();

        // null 체크와 기본값 설정
        String retVal = StringUtil.nvl(vop.get(REDIS_KEY_PREFIX + key), "empty");

        // 값이 존재하면 TTL 갱신 (슬라이딩 세션)
        if(!retVal.equals("empty")) {
            redisTemplate.expire(REDIS_KEY_PREFIX + key, 24, TimeUnit.HOURS);
        }

        return retVal;
    }

    /**
     * Key와 Value가 일치할 때만 삭제
     *
     * 왜 단순 삭제가 아닌가?
     *
     * 시나리오:
     * 1. 사용자 A가 PC에서 로그인 (sess_111)
     * 2. 사용자 A가 모바일에서 로그인 (sess_222)
     *    → Redis: user_A = sess_222 (덮어씀)
     * 3. PC에서 로그아웃 시도
     *    → 단순 삭제면: Redis에서 user_A 삭제됨
     *    → 문제: 모바일 세션까지 무효화됨!
     *
     * 해결:
     *    → 삭제 전에 값 비교
     *    → sess_111 ≠ sess_222 이므로 삭제 안 함
     *    → 모바일 세션은 유지됨
     */
    public static void removeByKeyAndValue(String key, String value) {
        ValueOperations<String, String> vop = redisTemplate.opsForValue();
        String redisVal = StringUtil.nvl(vop.get(REDIS_KEY_PREFIX + key), "empty");

        // 저장된 값과 일치할 때만 삭제
        if(redisVal.equals(value)) {
            redisTemplate.delete(REDIS_KEY_PREFIX + key);
        }
    }
}
```

### 1.4 LoginDupCheckInterceptor.java 분석

```java
/**
 * 파일: LoginDupCheckInterceptor.java
 * 위치: src/main/java/kr/co/mediopia/coss/hub/common/interceptor/
 *
 * 역할: 모든 HTTP 요청에서 중복 로그인 체크
 *
 * Spring Interceptor란?
 * → Controller 실행 전/후에 공통 로직 처리
 * → 필터와 유사하지만 Spring Context 내에서 동작
 */
@Component
public class LoginDupCheckInterceptor implements HandlerInterceptor {

    /**
     * Controller 실행 전 호출
     *
     * @return true: 계속 진행, false: 요청 중단
     */
    @Override
    public boolean preHandle(HttpServletRequest request,
                             HttpServletResponse response,
                             Object handler) throws Exception {

        // ========================================
        // Case 1: 로그인 안 된 사용자 → 통과
        // ========================================
        if (!SessionUtil.isLogin(request)) {
            return true;  // 검사 대상 아님
        }

        // ========================================
        // Case 2: 로그인된 사용자 → 중복 체크
        // ========================================

        HttpSession session = request.getSession();
        LoginInfo loginInfo = SessionUtil.getLoginInfo(session);

        // Redis에서 현재 유효한 세션 ID 조회
        String sessionId = RedisUtil.getValueByKey(loginInfo.getUserId());

        // ========================================
        // 세션 유효성 검사
        // ========================================

        // Case 2-1: Redis에 세션 정보 없음 (만료 또는 비정상)
        // Case 2-2: Redis 세션과 현재 세션 불일치 (다른 곳에서 로그인)
        if(sessionId.equals("empty") || !sessionId.equals(session.getId())) {

            // 로그아웃 처리
            SessionUtil.logout(session, request, response);

            // 메인 페이지로 리다이렉트
            response.sendRedirect("/");

            return false;  // 요청 중단
        }

        // Case 2-3: 정상 → 계속 진행
        return true;
    }

    // postHandle, afterCompletion은 기본 구현 사용
}
```

### 2.2 RedisComponent.java (공통) 분석

```java
/**
 * 파일: RedisComponent.java
 * 위치: src/main/java/egovframework/mediopia/lxp/common/comm/util/redis/
 *
 * 역할: 모든 모듈에서 사용하는 공통 Redis 컴포넌트
 *
 * 특징:
 * - 모듈명을 파라미터로 받아 키 네임스페이스 분리
 * - 다양한 데이터 타입 지원 (String, Hash, List)
 * - @Resource를 사용한 특정 타입 Operations 주입
 */
@Component
public class RedisComponent {

    // RedisTemplate 주입 (일반 연산용)
    @Autowired
    private RedisTemplate<String, Object> redisTemplate;

    // ===== 타입별 Operations 주입 =====
    // @Resource: 이름 기반 주입 (Spring의 @Autowired와 다름)

    // 문자열 연산
    @Resource(name = "redisTemplate")
    private ValueOperations<String, String> stringValOps;

    // 리스트 연산
    @Resource(name = "redisTemplate")
    private ListOperations<String, Object> listValOps;

    // 해시 연산
    @Resource(name = "redisTemplate")
    private HashOperations<String, String, Object> hashValOps;

    // 상수
    private static String REDIS_SITE = Constants.REDIS_SITE;        // "KNU10-LMS"
    private static long REDIS_DEFULT_EXPIRE = Constants.REDIS_DEFULT_EXPIRE;  // 3600

    // ===== 키 생성 =====

    /**
     * 모듈별 키 생성
     *
     * 형식: {사이트}:{모듈}:{키}
     *
     * 예시:
     *   parseKey("user123_course456", "lesson")
     *   → "KNU10-LMS:lesson:user123_course456"
     *
     * 장점:
     * - 모듈별 데이터 격리
     * - Redis CLI에서 패턴 검색 용이
     * - 키 충돌 방지
     */
    public String parseKey(String key, String moduleName) {
        return String.format("%s:%s:%s", REDIS_SITE, moduleName, key);
    }

    // ===== String 연산 =====

    /**
     * 문자열 저장
     *
     * @param key        저장할 키
     * @param value      저장할 값
     * @param expire     만료 시간 (초)
     * @param moduleName 모듈명
     */
    public void setString(String key, String value, long expire, String moduleName) {
        stringValOps.set(
            parseKey(key, moduleName),
            value,
            expire,
            TimeUnit.SECONDS
        );
    }

    /**
     * 문자열 저장 (기본 TTL 사용)
     */
    public void setString(String key, String value, String moduleName) {
        stringValOps.set(
            parseKey(key, moduleName),
            value,
            REDIS_DEFULT_EXPIRE,
            TimeUnit.SECONDS
        );
    }

    /**
     * 문자열 조회
     */
    public String getString(String key, String moduleName) {
        try {
            return (String) stringValOps.get(parseKey(key, moduleName));
        } catch (Exception e) {
            return null;  // 예외 시 null 반환 (조용한 실패)
        }
    }

    // ===== Hash 연산 =====

    /**
     * 해시 저장
     *
     * 사용 예시:
     *   Map<String, Object> userInfo = new HashMap<>();
     *   userInfo.put("name", "홍길동");
     *   userInfo.put("score", 95);
     *   putHash("exam123_user456", userInfo, 3600, "exam");
     *
     * Redis 결과:
     *   Key: "KNU10-LMS:exam:exam123_user456"
     *   Fields:
     *     name: "홍길동"
     *     score: 95
     */
    public void putHash(String key, Map<String, Object> value,
                        long expire, String moduleName) {
        hashValOps.putAll(parseKey(key, moduleName), value);
        redisTemplate.expire(parseKey(key, moduleName), expire, TimeUnit.SECONDS);
    }

    /**
     * 해시 특정 필드 조회
     */
    public Object getHash(String key, String hashKey, String moduleName) {
        try {
            return hashValOps.get(parseKey(key, moduleName), hashKey);
        } catch (Exception e) {
            return null;
        }
    }

    /**
     * 해시 필드 목록 조회
     */
    public Set<String> getHashKeys(String key, String moduleName) {
        try {
            return hashValOps.keys(parseKey(key, moduleName));
        } catch (Exception e) {
            return null;
        }
    }

    /**
     * 해시 특정 필드 삭제
     */
    public void deleteHash(String key, String moduleName, Object... keys) {
        hashValOps.delete(parseKey(key, moduleName), keys);
    }

    // ===== List 연산 =====

    /**
     * 리스트에 추가 (왼쪽)
     *
     * 사용 예시:
     *   pushList("notifications_user123", "새 알림!", 3600, "user");
     *
     * Redis 결과:
     *   Key: "KNU10-LMS:user:notifications_user123"
     *   Value: ["새 알림!", "이전 알림1", "이전 알림2", ...]
     */
    public void pushList(String key, Object value, long expire, String moduleName) {
        listValOps.leftPush(parseKey(key, moduleName), value);
        redisTemplate.expire(parseKey(key, moduleName), expire, TimeUnit.SECONDS);
    }

    /**
     * 리스트 전체 조회
     */
    public List<Object> getList(String key, String moduleName) {
        try {
            // 0부터 -1까지 = 전체 조회
            return listValOps.range(parseKey(key, moduleName), 0, -1);
        } catch (Exception e) {
            return null;
        }
    }

    // ===== 공통 연산 =====

    /**
     * 키 존재 여부 확인
     */
    public boolean isKey(String key, String moduleName) {
        return redisTemplate.hasKey(parseKey(key, moduleName));
    }

    /**
     * TTL 조회
     */
    public long getExpire(String key, String moduleName) {
        return redisTemplate.getExpire(parseKey(key, moduleName));
    }

    /**
     * 키 삭제
     */
    public void delete(String key, String moduleName) {
        redisTemplate.delete(parseKey(key, moduleName));
    }

    /**
     * TTL 갱신
     */
    public void refreshExpire(String key, long expire, String moduleName) {
        redisTemplate.expire(parseKey(key, moduleName), expire, TimeUnit.SECONDS);
    }
}
```

### 2.3 ExamRedisComponent.java (모듈 전용) 분석

```java
/**
 * 파일: ExamRedisComponent.java
 * 위치: src/main/java/egovframework/mediopia/lxp/lms/core/exam/service/redis/
 *
 * 역할: 시험(Exam) 모듈 전용 Redis 컴포넌트
 *
 * 특징:
 * - MODULE_NAME이 고정됨 ("exam")
 * - 모듈명 파라미터가 없어 사용이 간편
 * - 시험 관련 데이터만 처리
 *
 * 키 예시:
 * - "KNU10-LMS:exam:시험ID_학생ID"
 * - "KNU10-LMS:exam:answer_exam123_user456"
 */
@Component
public class ExamRedisComponent {

    @Autowired
    private RedisTemplate<String, Object> redisTemplate;

    @Resource(name = "redisTemplate")
    private ValueOperations<String, String> stringValOps;

    @Resource(name = "redisTemplate")
    private ListOperations<String, Object> listValOps;

    @Resource(name = "redisTemplate")
    private HashOperations<String, String, Object> hashValOps;

    private static String REDIS_SITE = Constants.REDIS_SITE;
    private static long REDIS_DEFULT_EXPIRE = Constants.REDIS_DEFULT_EXPIRE;

    // ★ 모듈명 고정 (RedisComponent와의 차이)
    private static String MODULE_NAME = "exam";

    /**
     * 키 생성 (모듈명 자동 적용)
     *
     * 비교:
     * - RedisComponent: parseKey("key", "exam") → 모듈명 필요
     * - ExamRedisComponent: parseKey("key") → 모듈명 불필요
     */
    public String parseKey(String key) {
        return String.format("%s:%s:%s", REDIS_SITE, MODULE_NAME, key);
    }

    // ===== 나머지 메서드는 RedisComponent와 동일한 패턴 =====
    // (모듈명 파라미터만 없음)

    public void setString(String key, String value, long expire) {
        stringValOps.set(parseKey(key), value, expire, TimeUnit.SECONDS);
    }

    public String getString(String key) {
        try {
            return (String) stringValOps.get(parseKey(key));
        } catch (Exception e) {
            return null;
        }
    }

    // ... (나머지 메서드 생략)
}
```

### 2.4 공통 vs 전용 컴포넌트 사용 가이드

!!! tip "공통 vs 전용 컴포넌트 선택 가이드"
    **RedisComponent (공통)**

    사용 시점: 여러 모듈에서 공통 사용 / 동적 모듈명 / 새 모듈 추가 시

    ```java
    @Autowired
    private RedisComponent redisComponent;

    // 모듈명을 파라미터로 전달
    redisComponent.setString("key", "value", "myModule");
    ```

    **ExamRedisComponent / LessonRedisComponent (전용)**

    사용 시점: 특정 모듈에서만 사용 / 고정 모듈명 / 타입 안전성 및 가독성 향상

    ```java
    @Autowired
    private ExamRedisComponent examRedis;

    // 모듈명 파라미터 불필요 (자동으로 "exam")
    examRedis.setString("key", "value");
    ```

    **권장:** 모듈이 명확하면 전용 컴포넌트 / 범용적 사용이면 공통 컴포넌트

### 3.1 Pub/Sub 전체 흐름

!!! note "Pub/Sub 전체 흐름도"
    ```mermaid

        participant Ctrl as ExamController
        participant Pub as ExamRedisPublisher
        participant Redis as Redis Server<br>Channel: exam_event

        participant Sub3 as WAS 3 Subscriber
        participant SSE as ExamSseService
        participant Student as 학생 브라우저

        Prof->>Ctrl: 시험 종료 버튼 클릭
        Ctrl->>Pub: publishStopEvent()
        Pub->>Redis: convertAndSend("exam_event", {eventType:"STOP", examId:"E123"})
        Redis->>Sub1: onMessage()
        Redis->>Sub2: onMessage()
        Redis->>Sub3: onMessage()
        Sub1->>SSE: stopRequest()
        Sub2->>SSE: stopRequest()
        Sub3->>SSE: stopRequest()
        SSE->>Student: "시험 종료됨"
    ```

### 3.2 Publisher 코드 분석

```java
/**
 * 파일: ExamRedisPublisher.java
 * 위치: src/main/java/.../lms/core/exam/service/redis/pubsub/
 *
 * 역할: Redis 채널로 시험 이벤트 발행
 *
 * 특징:
 * - 매우 단순한 구조 (한 메서드만 존재)
 * - RedisTemplate의 convertAndSend() 사용
 * - VO 객체를 직접 전송 (직렬화 자동 처리)
 */
@Service
public class ExamRedisPublisher {

    @Autowired
    private RedisTemplate<String, Object> redisTemplate;

    /**
     * 시험 이벤트 발행
     *
     * @param topic       발행할 채널 (예: ChannelTopic("exam_event"))
     * @param examPubSupVO 발행할 메시지 객체
     *
     * 내부 동작:
     * 1. examPubSupVO 객체를 직렬화
     * 2. topic 채널로 메시지 전송
     * 3. 해당 채널 구독자들이 메시지 수신
     */
    public void publish(ChannelTopic topic, ExamPubSupVO examPubSupVO) {
        // convertAndSend: 객체를 변환(직렬화)하여 전송
        redisTemplate.convertAndSend(topic.getTopic(), examPubSupVO);
    }
}
```

### 3.3 Subscriber 코드 분석

```java
/**
 * 파일: ExamRedisSubscriber.java
 * 위치: src/main/java/.../lms/core/exam/service/redis/pubsub/
 *
 * 역할: Redis 채널에서 시험 이벤트 수신 및 처리
 *
 * MessageListener 인터페이스:
 * - Redis 메시지를 수신하기 위한 표준 인터페이스
 * - onMessage() 메서드 구현 필요
 */
@Service
public class ExamRedisSubscriber implements MessageListener {

    @Autowired
    private RedisTemplate<String, Object> redisTemplate;

    @Autowired
    private ExamSseService examSseService;  // SSE 서비스 (브라우저 실시간 알림)

    /**
     * Redis 메시지 수신 콜백
     *
     * 이 메서드는 Redis에서 메시지를 수신할 때 자동으로 호출됨
     *
     * @param message Redis 메시지 객체 (바이트 배열)
     * @param pattern 채널 패턴 (패턴 구독 시 사용)
     */
    @Override
    public void onMessage(Message message, byte[] pattern) {

        // 메시지 본문이 있는지 확인
        if(redisTemplate.getValueSerializer().deserialize(message.getBody()) != null) {

            // ===== 메시지 역직렬화 =====
            // 바이트 배열 → Java 객체로 변환
            ExamPubSupVO pubSupVO = (ExamPubSupVO)
                redisTemplate.getValueSerializer().deserialize(message.getBody());

            // ===== 이벤트 타입별 분기 처리 =====
            switch (pubSupVO.getEventType()) {

                case "DUPLICATE_CHECK":
                    // 시험 중복 응시 체크
                    // → 같은 학생이 다른 기기에서 시험 응시 시도
                    examSseService.duplicateCheck(pubSupVO);
                    break;

                case "STOP":
                    // 시험 강제 종료
                    // → 교수가 시험 종료 버튼 클릭
                    // → 해당 시험 응시 중인 모든 학생에게 알림
                    examSseService.stopRequest(pubSupVO);
                    break;

                // 필요시 추가 이벤트 타입 처리
                // case "PAUSE":
                //     examSseService.pauseExam(pubSupVO);
                //     break;
            }
        }
    }
}
```

### 3.5 이벤트 타입별 사용 시나리오

!!! info "이벤트 타입별 시나리오"
    **[DUPLICATE_CHECK] - 시험 중복 응시 방지**

    1. 학생 A가 PC에서 시험 응시 중
    2. 학생 A가 모바일에서 같은 시험 접속 시도
    3. 서버에서 DUPLICATE_CHECK 이벤트 발행
    4. 모든 WAS가 수신
    5. PC의 시험 세션 종료 또는 모바일 접속 차단

    **[STOP] - 시험 강제 종료**

    1. 교수가 시험 종료 버튼 클릭 (예: 시간 연장 후 종료)
    2. STOP 이벤트 발행
    3. 모든 WAS가 수신
    4. 해당 시험 응시 중인 학생들에게 SSE 알림
    5. 학생 브라우저에서 답안 자동 제출

    **[DUPLICATE_LEARNING_CHECK] - 학습 중복 방지 (Lesson)**

    1. 학생이 PC에서 강의 수강 중
    2. 같은 학생이 모바일에서 같은 강의 재생 시도
    3. DUPLICATE_LEARNING_CHECK 이벤트 발행
    4. PC의 강의 재생 중지 또는 모바일 접속 차단

### 4.3 패턴 비교 및 권장

!!! tip "Static vs Component 패턴 비교"
    | 항목 | Static 유틸리티 | Component (Bean) |
    |------|----------------|-----------------|
    | 호출 방식 | `RedisUtil.get()` | `redisComponent.get()` |
    | 주입 필요 | 불필요 | 필요 (@Autowired) |
    | 테스트 | 어려움 (Mock 불가) | 쉬움 (Mock 가능) |
    | Spring | 필요 (초기화) | 완전 통합 |
    | 유연성 | 낮음 (고정) | 높음 (교체 가능) |
    | 코드량 | 적음 | 약간 많음 |

    **권장:**

    - 단순 유틸리티 → Static (HUB 방식)
    - 복잡한 로직, 테스트 필요 → Component (LMS 방식)
    - 새 프로젝트 → Component 권장 (Spring 표준)

## 5. Redis 사용 시 주의사항

### 5.1 일반적인 주의사항

!!! warning "Redis 사용 시 주의사항"
    **1. 메모리 관리**

    - Redis는 메모리에 저장 → 용량 제한 있음
    - TTL 설정 필수 (만료 없으면 메모리 누수)
    - 주기적인 메모리 모니터링 필요

    **2. 데이터 영속성**

    - 기본적으로 메모리 저장 → 재시작 시 데이터 손실 가능
    - 중요 데이터는 DB에도 저장
    - RDB/AOF 영속화 설정 고려

    **3. 연결 관리**

    - 연결 풀 사이즈 적절히 설정
    - 연결 타임아웃 설정
    - 연결 끊김 시 재연결 로직 필요

    **4. 직렬화**

    - 클래스 변경 시 역직렬화 오류 가능
    - JSON 직렬화 권장 (호환성)
    - 대용량 객체 직렬화 시 성능 저하

    **5. 키 네이밍**

    - 환경별 접두어 사용 (충돌 방지)
    - 의미 있는 키 이름 사용
    - 너무 긴 키 이름 피하기 (메모리 낭비)

### 5.2 코드 레벨 주의사항

```java
// ❌ 잘못된 예시
public void badExample() {

    // 1. TTL 없이 저장 → 메모리 누수
    redisTemplate.opsForValue().set("key", "value");

    // 2. 예외 처리 없음 → 장애 전파
    String value = (String) redisTemplate.opsForValue().get("key");

    // 3. 키 중복 가능성
    redisTemplate.opsForValue().set("user", "data");  // 어떤 user?

    // 4. 무한 루프 가능
    while(redisTemplate.opsForValue().get("key") == null) {
        // 영원히 null일 수 있음
    }
}

// ✅ 올바른 예시
public void goodExample() {

    // 1. TTL 설정
    redisTemplate.opsForValue().set("key", "value", 1, TimeUnit.HOURS);

    // 2. 예외 처리
    String value;
    try {
        value = (String) redisTemplate.opsForValue().get("key");
    } catch (Exception e) {
        log.error("Redis 조회 실패", e);
        value = null;  // 기본값 처리
    }

    // 3. 명확한 키 네이밍
    String key = String.format("KNU10-LMS:user:%s:session", userId);
    redisTemplate.opsForValue().set(key, "data");

    // 4. 재시도 횟수 제한
    int retryCount = 0;
    while(redisTemplate.opsForValue().get("key") == null && retryCount < 3) {
        Thread.sleep(100);
        retryCount++;
    }
}
```

## 6. 실무 Best Practice

### 6.1 키 네이밍 컨벤션

!!! info "키 네이밍 컨벤션"
    **권장 형식:** `{환경}:{서비스}:{모듈}:{엔티티}:{ID}`

    **예시:**

    ```
    prod:lms:exam:answer:exam123_user456
    │    │   │    │      └── 식별자
    │    │   │    └──────── 엔티티 (답안)
    │    │   └───────────── 모듈 (시험)
    │    └───────────────── 서비스 (LMS)
    └────────────────────── 환경 (운영)
    ```

    **현재 프로젝트 패턴:**

    - HUB: `KNU10-HUB:userId`
    - LMS: `KNU10-LMS:module:key`

    **개선 제안:**

    - `KNU10-HUB-PROD:session:userId`
    - `KNU10-LMS-PROD:exam:answer:examId_userId`

### 6.2 TTL 설정 가이드

!!! note "TTL 설정 가이드"
    **데이터 유형별 권장 TTL:**

    | 데이터 유형 | 권장 TTL | 이유 |
    |------------|----------|------|
    | 로그인 세션 | 24시간 | 일일 작업 주기 |
    | API 캐시 | 5~30분 | 데이터 신선도 |
    | OTP 인증 | 3~5분 | 보안상 짧게 |
    | Rate Limiting | 1분 | 분당 제한 |
    | 시험 세션 | 3시간 | 시험 최대 시간 + 여유 |
    | 임시 데이터 | 1시간 | 적당한 유지 |
    | 설정/메타데이터 | 없음/긴 시간 | 거의 변경 안 됨 |

    **슬라이딩 세션 적용 기준:**

    - 활동 중 만료되면 안 되는 경우 → 슬라이딩 적용
    - 절대 만료 시간이 필요한 경우 → 고정 TTL

### 6.3 에러 처리 패턴

```java
/**
 * Redis 에러 처리 Best Practice
 */
@Component
public class ResilientRedisComponent {

    @Autowired
    private RedisTemplate<String, Object> redisTemplate;

    private static final Logger log = LoggerFactory.getLogger(ResilientRedisComponent.class);

    /**
     * 안전한 Redis 조회
     *
     * 특징:
     * 1. 예외 시 null 반환 (장애 전파 방지)
     * 2. 에러 로깅
     * 3. 호출부에서 null 체크 필요
     */
    public String safeGet(String key) {
        try {
            return (String) redisTemplate.opsForValue().get(key);
        } catch (Exception e) {
            log.error("Redis GET 실패. key={}", key, e);
            return null;  // 기본값 반환
        }
    }

    /**
     * 기본값 반환 패턴
     */
    public String getOrDefault(String key, String defaultValue) {
        try {
            String value = (String) redisTemplate.opsForValue().get(key);
            return value != null ? value : defaultValue;
        } catch (Exception e) {
            log.warn("Redis 조회 실패, 기본값 반환. key={}", key);
            return defaultValue;
        }
    }

    /**
     * Fallback 패턴 (Redis 실패 시 DB 조회)
     */
    public String getWithFallback(String key, Supplier<String> dbFallback) {
        try {
            String value = (String) redisTemplate.opsForValue().get(key);
            if (value != null) {
                return value;
            }
        } catch (Exception e) {
            log.warn("Redis 조회 실패, DB Fallback 실행. key={}", key);
        }

        // Redis 실패 또는 캐시 미스 → DB 조회
        String dbValue = dbFallback.get();

        // 가능하면 Redis에 다시 저장
        try {
            if (dbValue != null) {
                redisTemplate.opsForValue().set(key, dbValue, 1, TimeUnit.HOURS);
            }
        } catch (Exception e) {
            log.warn("Redis 저장 실패. key={}", key);
        }

        return dbValue;
    }
}

// 사용 예시
@Service
public class UserService {

    @Autowired
    private ResilientRedisComponent redisComponent;

    @Autowired
    private UserRepository userRepository;

    public String getUserName(String userId) {
        return redisComponent.getWithFallback(
            "user:" + userId + ":name",
            () -> userRepository.findNameById(userId)  // DB Fallback
        );
    }
}
```
